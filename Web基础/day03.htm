<html>

  <head>

    <title>TEST1.html</title>

        

    <meta http-equiv="keywords" content="keyword1,keyword2,keyword3">

    <meta http-equiv="description" content="this is my page">

    <meta http-equiv="content-type" content="text/html; charset=UTF-8">

   

    <!--<link rel="stylesheet" type="text/css" href="./styles.css">-->

         <script type="text/javascript">

        

         var number ;

        

                   function guess(){

                           

                            //1 生成一个0~100之间的随机整数

                            if(number == undefined){//如果number没有初始化值，就赋值

                                     number = Math.round(Math.random()*100);

                            //random产生的随机数在0~1之间，所以要乘以100，再舍去小数，所以用                         //round

                            }

                           

                            //2 弹出输入框要求用户输入

                            var input = prompt("请您输入一个0~100之间的整数","0");

                           

                            //3 判断用户输入的数字

                            if(isNaN(+input)){//将字符串转化为数字并判断是否为NaN

                                               // 不是数字=>提示重新输入有效数字=>重复23步

                                     alert("请输入一个数字!");

                                     guess();//递归调用，重复前面步骤

                                    

                            }else if(input>number){

                                               //  大了=> 提示=>重复23步

                                     alert("大了!");

                                     guess();

                            }else if(input<number){

                                     // 小了 => 提示=>重复23步

                                     alert("小了!");

                                     guess();

                                    

                            }else{

                                     number=undefined;

                                               // 猜对了=> 恭喜并询问用户是否再次游戏

                                     var result = confirm("恭喜您猜对了!是否需要再次游戏?");      

                                     if(result){

                                               //再次游戏 => 重复123

                                               guess();

                                     }

                                               //不再游戏=> 结束

                                              

                            }

                   }

        

         </script>

案例1：节点的增删改查

<html>

<head>

<title>节点的增删改查</title>

<meta http-equiv="Content-Type" content="text/html; charset=utf-8" />

 

<!--加入样式表-->

<style type="text/css">

 

div {

         border:1px red solid;

         height: 60px;

         width: 120px;

         margin: 20px 0px 20px 20px;

         padding: 10px 0px 0px 20px;

}

 

#div_1 {

         background-color: #00FFFF;

}

 

#div_2 {

         background-color: #FF3399;

}

 

#div_3 {

         background-color: #0000FF;

}

 

#div_4 {

         background-color: #FFFF66;

}

 

</style>

</head>

<body>

 

         <div id="div_1"></div>

 

         <div id="div_2">区域2</div>

 

         <div id="div_3">区域3</div>

 

         <div id="div_4">区域4</div>

 

         <hr/>

        

         <input type="button" value="添加节点" onclick="addNode()" />

         <input type="button" value="删除节点" onclick="deleteNode()" />

         <input type="button" value="替换节点" onclick="updateNode()" />

         <input type="button" value="克隆节点" onclick="copyNode()" />

 

</body>

<script type="text/javascript">

 

         //创建并添加节点

         function addNode() {

                   // 1 创建A标签对象，创建出来的是空的：<a></a>

                   var a = document.createElement("a");

                  

                   // 2 为创建的a标签增加属性

                   a.setAttribute("href", "http://www.itcast.cn");

                  

                   // 3 为a标签添加标签体

                   a.innerHTML = "点我";

                  

                   // 4 找到div标签

                   var div_1 = document.getElementById("div_1");

                  

                   // 5 添加

                   div_1.appendChild(a);

 

         }

        

         //删除节点

         function deleteNode() {

                   //1 获得要删除的div

                   var div_2 = document.getElementById("div_2");

                  

                   //2 获得div的父元素

                   var parent = div_2.parentNode;

                  

                   //3 通过爹删除孩子

                   parent.removeChild(div_2);

                  

         }

 

         //替换节点

         function updateNode() {

                   //1 获得被替换的div

                   var div_3 = document.getElementById("div_3");

                  

                   //2 创建img元素对象<img />

                   var img = document.createElement("img");

                  

                   //3 为img添加属性

                            function fun1(){

                                     flag =  false;

                            //验证用户名，要使用的时候直接在上面的fun1方法中调用即可

                            //1 用户名不能为空,并且长度在6到10位之间.只能以英文字母开头. 符号上                     //用户名中只能使用"_".

 

                            function checkName(){

 

                            //1 获得文本输入框中的内容

                            var nameInput=document.getElementsByName("name")[0];                                  

                            var name = nameInput.value;  //获得属性值

                                             

                            //2 定义正则表达式对象

                            var reg = /^[a-zA-Z][a-zA-Z0-9_]{5,9}$/g;

                            //正则表达式填了肯定不为空；^[a-zA-Z]中，^表示首位，[a-zA-Z]表示只能为                       英文字母；

                            //[a-zA-Z0-9_]{5,9}中，表示5~9位是英文字母数字和下划线，所以加起来就是                     //6~10位；

                            //实际开发中应使用工具自动生成正则表达式

                                    

                            //3 校验

                                     if(!reg.test(name)){//检验成功返回true，检验失败返回false

                                               //校验失败=> 提示用户.

                                               addError(nameInput,"用户需要6到10位,不能以数字开头!");

                                               //addError方法见后

                                               return false;//checkName方法返回false

                                     }else{

                                               //验证成功=>清除错误信息

                                               removeError(nameInput);

                                               //removeError方法见后

                                               return true;//checkName方法返回true

                                     }

                                     }

                            var email = document.form1.email.value;//注意这种简便方法

                           

                            //2 定义正则

                            var reg = /^([a-z0-9_\.-]+)@([\da-z\.-]+)\.([a-z\.]{2,6})$/g;

                            //使用工具自动生成。

                           

                            //3 校验

                            if(!reg.test(email)){

                                     //失败=>提示,并返回false

                                               return false;

                                               //checkEmail方法返回false

 

                            }else{//验证成功，清除错误信息

                                     removeError(document.form1.email);

                                     return true;

                                     //checkEmail方法返回true

                            }       

                            }

 

                            //添加错误信息的方法

                            //参数1 : 信息往哪里添加

                            //参数2 : 错误信息是什么

                                     function addError(where,what){

                                               where.nextSibling.innerHTML=what;

                                               //添加错误信息的地方应该在输入框的右侧，

                                               //在上面的html代码中，输入框右侧是font标签,

                                               //所以where表示输入框，则nextSibling就表示font标签

                                               //font标签的值就是方法传进来的what

                            }

                            //第二次填写正确，就应该把第一次出现的错误信息清除

                            function removeError(where){

                                     where.nextSibling.innerHTML="";

                            }
