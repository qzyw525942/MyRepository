public class AServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   // 1 获得Application域中存放的统计数字

                   Integer count = (Integer) getServletContext().getAttribute("count");

 

                   // 2 判断是否得到统计数字

                   if (count == null)

                   {

                            // 没得到=> 将数字初始化为1

                            count = 1;

                   } else

                   {

                            // 得到了=> 将数字加1

                            count += 1;

                   }

 

                   // 3 输出,并放回到Application域中

                   response.getWriter().write("you are the " + count + " vistors");

 

                   getServletContext().setAttribute("count", count);

//逻辑颠倒，先获得键值对，再设置键值对

         }

 

}

 

只要其它Servlet中有与上面Servlet的内容一样的内容，则不管访问哪个servlet，都能得到网站访问人次的数值。

案例5：加了积分要求的下载

 

一般的开放式下载无需什么东西直接下载就行，但有的下载必须有积分之类的东西。前者可用html代码直接实现，后者需要servlet控制。

 

public class FServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   // 1 获得图片的输入流

                   InputStream is = getServletContext().getResourceAsStream(

                                     "/WEB-INF/xxx1.jpg");

 

                   // 2 获得输出到浏览器的输出流

                   OutputStream os = response.getOutputStream();

 

                   // 3 两个流对接

                   byte[] bys = new byte[1024];

                   int len = -1;

 

                   while ((len = is.read(bys)) != -1)

                   {

                            os.write(bys, 0, len);

                            os.flush();// 字节流也可以刷新一下，不然图片不完整

                   }

 

                   os.close();

                   is.close();

 

//4、输入输出流都控制在手中，就好比掌握了开关一样，给积分就下载，不给就    //不让下。

 

         }

 

}

 

案例6：文件下载

 

public class GServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   // 0 原则: 凡是响应正文中需要输出内容, 一定要设置content-type头

                   ServletContext sc = getServletContext();

                   String type = sc.getMimeType("abc.zip");// 前面的名字随便写，后缀名要写对，表示                                                                                     //要下载的文件的类型

                   response.setContentType(type);

 

                   // 0 设置响应头:提示用户文件要以什么名称保存 => Http协议不支持中文码表，所                   //以要手动设置中文名称的文件的编码方式

                   // URLEncoder.encode("apache-汤姆-6.0.35.zip","UTF-8");

 

                   response.setHeader("Content-Disposition", "attachment;filename="

                                     + URLEncoder.encode("apache-汤姆-6.0.35.zip", "UTF-8"));

 

                   // 1 获得文件的输入流

                   InputStream is = getServletContext().getResourceAsStream(

                                     "/WEB-INF/apache-tomcat-6.0.35.zip");

 

                   // 2 获得输出到浏览器的输出流

                   OutputStream os = response.getOutputStream();

 

                   // 3 两个流对接

                   byte[] buffer = new byte[1024];

                   int len = -1;

 

                   while ((len = is.read(buffer)) != -1)

                   {

                            os.write(buffer, 0, len);

                            os.flush();

                   }

 

                   os.close();

                   is.close();

 

         }

 

}

public class BServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   // 1 获得 所有参数的键

                   Enumeration<String> en = request.getParameterNames();

 

                   // 2 遍历所有键

                   while (en.hasMoreElements())

                   {

                            String key = en.nextElement();

 

                            // 根据键获得值

                            String value = request.getParameter(key);

 

                            System.out.println(key + "==>" + value);

                   }

         }

 

         @Override

         protected void doPost(HttpServletRequest req, HttpServletResponse resp)

                            throws ServletException, IOException

         {

                   // 1 获得 所有参数的键

                   Enumeration<String> en = req.getParameterNames();

 

                   // 2 遍历所有键

                   while (en.hasMoreElements())

                   {

                            String key = en.nextElement();

 

                            // 根据键 获得值

                            String value = req.getParameter(key);

 

                            System.out.println(key + "==>" + value);

                   }

         }

 

}

             用户名:<input type="text" name="name" /><br/>

             密码:<input type="password" name="password" /><br/>

             <input type="submit" value="提交" />

    </form>

   

  </body>

 

servlet文件：

public class CServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   String name = request.getParameter("name");

                   System.out.println(name);// 表单中输入中文可见name为乱码

 

                   // 1将name使用ISO-8859-1码表编码回字节数组。服务器默认使用的是ISO-8859-1

byte[] bytes = name.getBytes("ISO-8859-1");

 

                   // 2 将字节用UTF-8码表解码回中文

                   String name2 = new String(bytes, "UTF-8");

                   System.out.println(name2);

 

         }

 

         @Override

         protected void doPost(HttpServletRequest request, HttpServletResponse resp)

                            throws ServletException, IOException

         {

 

                   // post解决乱码,只要在调用获得参数的方法前.设置解码码表即可

                   request.setCharacterEncoding("UTF-8");

 

                   String name = request.getParameter("name");

                   System.out.println(name);

         }

 

}

B、request域：

把一个servlet处理后的结果封装在域中，从而可以从另一个servlet中获得这些结果继续处理。使用于请求转发中。

 

DServlet：

public class DServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   String name = "肉丝";

                   // 假设该值是从数据库中取出来.要将该值交给EServlet显示

 

                   // 将name变量放入request域：

                   // application，request和session 三大域的方法都是一样的

                   request.setAttribute("name", name);

 

//response.getWriter().write("DServlet hello!");

//不能在转发的servlet中向浏览器响应输出任何内容。被转发的servlet中可以

//在此代码之后不允许再对request, response进行操作。

C、请求包含

 

 

比如很多个页面的底部都相同，就可以把这些相同的部分封装到一个JSP中去，其他的部分每个页面一个JSP。在设计这些相同的部分时，就可以把封装了相同部分的JSP引入进来。这时就可以使用请求包含。

 

但是JSP还没学，所以这里仍用servlet来代替：

 

//FServlet :

public class FServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   response.setContentType("text/html;charset=utf-8");

 

                   response.getWriter().write("这是第一部分!<br>");

 

                   // 包含

                   request.getRequestDispatcher("/GServlet").include(request, response);

 

                   response.getWriter().write("这是第二部分!<br>");

                   // 包含之后还可以对request, response二者进行操作

         }

 

}

 

//GServlet

public class GServlet extends HttpServlet

{

 

         public void doGet(HttpServletRequest request, HttpServletResponse response)

                            throws ServletException, IOException

         {

 

                   response.getWriter().write("这是相同的部分!<br>");

                   // 两个servlet都可向浏览器返回数据

 

         }
