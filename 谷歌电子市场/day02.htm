第一集  ListView的封装

 

每次在写ListView的时候，都是那几个步骤，次数写多了很繁琐。现在来封装ListView，使它使用起来更容易。ListView的封装指的是其adapter的封装。

 

/**

 * 第一步：自定义ListView的Adapter类，继承自BaseAdapter；通过此类的构造方法把外界的集合数据传递进来，其中涉及到泛型的使用；

 * 重写getCount，getItem和getItemId三个方法

 */

public abstract class MyBaseAdapter<T> extends BaseAdapter

{

 

        // 注意: 此处必须要从0开始写

        private static final int TYPE_NORMAL = 0;// 正常布局类型

        private static final int TYPE_MORE = 1;// 加载更多类型

 

        private ArrayList<T> data;

 

        public MyBaseAdapter(ArrayList<T> data)

        {

                this.data = data;

        }

 

        @Override

        public int getCount()

        {

                return data.size() + 1;// 增加加载更多布局的数量

        }

 

        @Override

        public T getItem(int position)

        {

                return data.get(position);

        }

 

        @Override

        public long getItemId(int position)

        {

                return position;

        }

 

        /**

         * 第二步：getView的封装:见BaseHolder类

         */

 

        /**

         * 第七步：重写加载更多布局的getViewTypeCount方法和getItemViewType方法

         */

        // 返回lv布局类型的种类数

        @Override

        public int getViewTypeCount()

        {

                return 2;// 返回两种类型,普通布局以及加载更多类型布局

        }

 

        // 返回当前位置应该展示哪种布局类型

        @Override

        public int getItemViewType(int position)

        {

                if (position == getCount() - 1)

                {// 最后一个

 

                        return TYPE_MORE;

 

                } else

                {

                        return getInnerType();

                }

        }

 

        // 这样写的话，子类可以重写此方法来更改返回的布局类型。因为除了普通布局和加载更多之外，可能子类还需要其他布局

        // 注意，这个方法是自定义的，不是BaseAdapter提供的。

        public int getInnerType()

        {

                return TYPE_NORMAL;// 默认就是普通类型，当然子类重写此方法后可以修改此方法体

        }

 

        /**

         * 第八步：getView的具体实现

         */

        @Override

        public View getView(int position, View convertView, ViewGroup parent)

        {

 

                /**

                 * 第九步：holder的创建

                 */

                BaseHolder holder;

                if (convertView == null)

                {

                        // 这里相当于完成了如下三步：

                        // 1. 加载布局文件

                        // 2. 初始化控件 findViewById

                        // 3. 设置tag

                        if (getItemViewType(position) == TYPE_MORE)

                        {

                                // 加载更多的类型

                                holder = new MoreHolder(hasMore());// 返回具体MoreHolder对象

 

                        } else

                        {

                                holder = getHolder();// 返回具体BaseHolder对象

                        }

 

                } else

                {

                        holder = (BaseHolder) convertView.getTag();// 从convertView中去获取holder对象

                }

 

                /**

                 * 第十步：根据数据来刷新界面

                 */

                if (getItemViewType(position) != TYPE_MORE)

                {

                        // 普通布局

                        holder.setData(data.get(position));

 

                } else

                {

 

                        /**

                         * 第十一步：创建MoreHolder类；11~15步见MoreHolder类

                         */

 

                        /**

                         * 第十八步： 加载更多数据

                         */

                        // 一旦加载更多布局展示出来, 就开始加载更多

                        // 只有在有更多数据的状态下才加载更多

                        MoreHolder moreHolder = (MoreHolder) holder;

                        if (moreHolder.getData() == MoreHolder.STATE_MORE_MORE)

                        {

                                loadMore(moreHolder);

                        }

                }

 

                return holder.getRootView();

        }

 

        // 返回当前页面的holder对象, 必须由子类去实现

        public abstract BaseHolder<T> getHolder();

 

        // 子类可以重写此方法来决定是否可以加载更多。原理与getInnerType方法一样

        public boolean hasMore()

        {

                return true;// 默认都是有更多数据的

        }

 

        /**

         * 第十六步： 访问网络，加载更多数据

         */

        private boolean isLoadMore = false;// 标记是否正在加载更多

 

        public void loadMore(final MoreHolder holder)

        {

                if (!isLoadMore)

                {

                        isLoadMore = true;

 

                        new Thread()

                        {

                                @Override

                                public void run()

                                {

                                        final ArrayList<T> moreData = onLoadMore();

 

                                        /**

                                         * 第十七步： 主线程刷新界面

                                         */

                                        UIUtils.runOnUIThread(new Runnable()

                                        {

 

                                                @Override

                                                public void run()

                                                {

                                                        if (moreData != null)

                                                        {

 

                                                                // 每一页有20条数据, 如果返回的数据小于20条,

                                                                // 就认为到了最后一页了，就认为没有更多数据了

                                                                if (moreData.size() < 20)

                                                                {

                                                                        holder.setData(MoreHolder.STATE_MORE_NONE);

 

                                                                        Toast.makeText(UIUtils.getContext(),

                                                                                        "没有更多数据了", 1).show();

                                                                } else

                                                                {

                                                                        // 还有更多数据

                                                                        holder.setData(MoreHolder.STATE_MORE_MORE);

                                                                }

 

                                                                // 将更多数据追加到当前集合中

                                                                data.addAll(moreData);

 

                                                                // 刷新界面

                                                                MyBaseAdapter.this.notifyDataSetChanged();

 

                                                        } else

                                                        {

                                                                // 加载更多失败

                                                                holder.setData(MoreHolder.STATE_MORE_ERROR);

                                                        }

 

                                                        isLoadMore = false;

                                                }

                                        });

                                }

                        }.start();

                }

 

        }

 

        // 加载更多数据, 必须由子类实现

        public abstract ArrayList<T> onLoadMore();

 

}

//getView的封装需要使用到ViewHolder,而每个ListView条目的ViewHolder又是不一样的；所以找到它们的共性，抽取为此BaseHolder

public abstract class BaseHolder<T>

{

 

        private View mRootView;

        private T data;

 

        // 当new这个对象时, 就会加载布局初始化控件;设置tag

        public BaseHolder()

        {

                mRootView = initView();

 

                /**

                 * 第四步: 为item布局设置Holder:本类就是Holder

                 */

                mRootView.setTag(this);

        }

 

        /**

         * 第三步: 加载lv的item的布局并初始化控件

         * findViewById；在本类的构造方法中调用此方法；并提供返回item布局的getRootView方法

         * 。本类的几个步骤都是按照getView方法的书写顺序进行书写的

         */

        public abstract View initView();

 

        // 返回item的布局对象

        public View getRootView()

        {

                return mRootView;

        }

 

        /**

         * 第六步:设置和获取item的数据，第七步见MyBaseAdapter类

         */

        // 设置当前item的数据

        public void setData(T data)

        {

                this.data = data;

 

                refreshView(data);

        }

 

        // 获取当前item的数据

        public T getData()

        {

                return data;

        }

 

        /**

         * 第五步:根据数据来刷新界面

         */

        public abstract void refreshView(T data);

 

}

 

 

MoreHolder类如下：

/**

 * 第十一步：创建MoreHolder类，继承自BaseHolder

 */

public class MoreHolder extends BaseHolder<Integer>

{

 

        /**

         * 第十二步： 分析加载更多的几种状态

         */

        // 1. 可以加载更多

        // 2. 加载更多失败

        // 3. 没有更多数据

        public static final int STATE_MORE_MORE = 1;

        public static final int STATE_MORE_ERROR = 2;

        public static final int STATE_MORE_NONE = 3;

 

        private LinearLayout llLoadMore;

        private TextView tvLoadError;

 

        /**

         * 第十三步： 构造方法传递状态

         */

        public MoreHolder(boolean hasMore)

        {

                // 如果有更多数据,状态为STATE_MORE_MORE,否则为STATE_MORE_NONE,将此状态传递给父类的 T data,

                // 父类会同时刷新界面

                // setData方法来自父类

                setData(hasMore ? STATE_MORE_MORE : STATE_MORE_NONE);

                // setData结束后一定会调refreshView.后者的参数也就是这里传递的值

        }

 

        /**

         * 第十四步： 加载更多的界面。如下：

         */

        @Override

        public View initView()

        {

                View view = UIUtils.inflate(R.layout.list_item_more);

 

                llLoadMore = (LinearLayout) view.findViewById(R.id.ll_load_more);

                tvLoadError = (TextView) view.findViewById(R.id.tv_load_error);

 

                return view;

        }

 

        /**

         * 第十五步： refreshView中根据状态判断加载更多布局是否显示隐藏

*16步之后见MyBaseAdapter类

         */

        @Override

        public void refreshView(Integer data)

        {

                switch (data)

                {

                case STATE_MORE_MORE:

 

                        // 显示加载更多

                        llLoadMore.setVisibility(View.VISIBLE);

                        tvLoadError.setVisibility(View.GONE);

 

                        break;

 

                case STATE_MORE_NONE:

 

                        // 隐藏加载更多

                        llLoadMore.setVisibility(View.GONE);

                        tvLoadError.setVisibility(View.GONE);

 

                        break;

 

                case STATE_MORE_ERROR:

 

                        // 显示加载失败的布局

                        llLoadMore.setVisibility(View.GONE);

                        tvLoadError.setVisibility(View.VISIBLE);

 

                        break;

 

                default:

                        break;

                }

        }

 

}

第二集  网络访问的封装

 

把演示项目的HttpClientFactory和HttpHelper两个类复制到自己项目；并把其中所需要的几个工具类也复制过来。

 

/**

 * 第一步：创建访问网络的基类.Protocol：协议

* 这里的T是数据解析之后生成的对象类

 */

public abstract class BaseProtocol<T>

{

 

        /**

         * 第七步： 获取数据并解析

         */

        // index表示的是从哪个位置开始返回20条数据, 用于分页

        public T getData(int index)

        {

                // 先判断是否有缓存, 有的话就加载缓存

                String result = getCache(index);

 

                if (StringUtils.isEmpty(result))

                {// 如果没有缓存,或者缓存失效，才请求服务器

                        result = getDataFromServer(index);

                }

 

                // 开始解析

                if (result != null)

                {

                        T data = parseData(result);

 

                        return data;

                }

 

                return null;

        }

 

        /**

         * 第二步：从网络获取数据

         */

        private String getDataFromServer(int index)

        {

                // URL类似于：

                // http://www.itheima.com/home?index=0&name=zhangsan&age=18

                // index表示的是从哪个位置开始返回20条数据, 用于分页

                HttpResult httpResult = HttpHelper.get(HttpHelper.URL + getKey()

                                + "?index=" + index + getParams());

 

                if (httpResult != null)

                {

                        String result = httpResult.getString();

 

                        System.out.println("访问结果:" + result);

 

                        /**

                         * 第五步： 把网络数据写入缓存

                         */

                        if (!StringUtils.isEmpty(result))

                        {

                                setCache(index, result);

                        }

 

                        return result;

                }

 

                return null;

        }

 

        // 获取网络链接关键词 /home, 子类必须实现

        public abstract String getKey();

 

        // 获取网络链接参数, 子类必须实现

        public abstract String getParams();

 

        /**

         * 第三步： 写缓存。这里的读写缓存是标准的作法。

         */

        // 之前存入sp中是 以url为key, 以json为value。现在以url为文件名, 以json为文件内容,保存在本地

        public void setCache(int index, String json)

        {

 

                File cacheDir = UIUtils.getContext().getCacheDir();

                // 本项目的缓存文件夹。

 

                // 生成缓存文件

                File cacheFile = new File(cacheDir, getKey() + "?index=" + index

                                + getParams());

                // 第一个参数是缓存文件夹，第二个参数是url，也就是文件名

 

                FileWriter writer = null;

                try

                {

                        writer = new FileWriter(cacheFile);

 

                        // 缓存失效的截止时间

                        long deadline = System.currentTimeMillis() + 30 * 60 * 1000;

// 当前时间+将来半个小时的有效期

                        writer.write(deadline + "\n");// 在第一行写入缓存时间, 换行

 

                        writer.write(json);// 写入json数据

                        writer.flush();

 

                } catch (IOException e)

                {

                        e.printStackTrace();

 

                } finally

                {

                        IOUtils.close(writer);

                }

        }

 

        /**

         * 第四步： 读缓存

         */

        // 以url为文件名, 以json为文件内容,保存在本地

        public String getCache(int index)

        {

                File cacheDir = UIUtils.getContext().getCacheDir();// 本应用的缓存文件夹

                // 生成缓存文件

                File cacheFile = new File(cacheDir, getKey() + "?index=" + index

                                + getParams());

 

                // 判断缓存是否存在

                if (cacheFile.exists())

                {

                        // 判断缓存是否有效

                        BufferedReader reader = null;

                        try

                        {

                                reader = new BufferedReader(new FileReader(cacheFile));

                                String deadline = reader.readLine();// 读取第一行的有效期

                                long deadtime = Long.parseLong(deadline);

 

                                if (System.currentTimeMillis() < deadtime)

                                {// 当前时间小于截止时间,

                                        // 说明缓存有效

 

                                        StringBuffer sb = new StringBuffer();

                                        String line;

                                        while ((line = reader.readLine()) != null)

                                        // 这里是继第一行之后继续往后读，即从第二行开始往

//后读，所以无需去掉第一行

                                        {

                                                sb.append(line);

                                        }

 

                                        return sb.toString();

                                }

 

                        } catch (Exception e)

                        {

                                e.printStackTrace();

 

                        } finally

                        {

                                IOUtils.close(reader);                   

                        }

 

                }

 

                return null;

        }

 

        /**

         * 第六步：解析json数据, 子类必须实现

         */

        public abstract T parseData(String result);

}

/**

 * 首页网络数据解析.继承自BaseProtocol需要实现如下三个方法

 */

public class HomeProtocol extends BaseProtocol<ArrayList<AppInfo>>

{

 

        @Override

        public String getKey()

        {

                return "home";

        }

 

        @Override

        public String getParams()

        {

                return "";// 如果没有参数,就传空串,不要传null

        }

 

        @Override

        public ArrayList<AppInfo> parseData(String result)

        {

                // 使用JsonObject解析方式: 如果遇到{},就是JsonObject;如果遇到[], 就是JsonArray

                try

                {

                        JSONObject jo = new JSONObject(result);

 

                        // 解析应用列表数据

                        JSONArray ja = jo.getJSONArray("list");

 

                        ArrayList<AppInfo> appInfoList = new ArrayList<AppInfo>();

 

                        for (int i = 0; i < ja.length(); i++)

                        {

                                JSONObject jo1 = ja.getJSONObject(i);

 

                                AppInfo info = new AppInfo();

                                // AppInfo见下：

                                /*

                                public class AppInfo

                                {

 

                                        public String des;

                                        public String downloadUrl;

                                        public String iconUrl;

                                        public String id;

                                        public String name;

                                        public String packageName;

                                        public long size;

                                        public float stars;

 

                                }*/

 

 

                                info.des = jo1.getString("des");

                                info.downloadUrl = jo1.getString("downloadUrl");

                                info.iconUrl = jo1.getString("iconUrl");

                                info.id = jo1.getString("id");

                                info.name = jo1.getString("name");

                                info.packageName = jo1.getString("packageName");

                                info.size = jo1.getLong("size");

                                info.stars = (float) jo1.getDouble("stars");

 

                                appInfoList.add(info);

                        }

 

                        // 初始化轮播条的数据

                        JSONArray ja1 = jo.getJSONArray("picture");

                       

                        /*  "picture": ["image/home01.jpg"]  */

                       

                        ArrayList<String> pictures = new ArrayList<String>();

                       

                        for (int i = 0; i < ja1.length(); i++)

                        {

                                String pic = ja1.getString(i);

                               

                                pictures.add(pic);

                        }

 

                        return appInfoList;

 

                } catch (JSONException e)

                {

                        e.printStackTrace();

                }

 

                return null;

        }

 

}

public class HomeFragment extends BaseFragment

{

 

        private ArrayList<AppInfo> data;

 

        // 如果加载数据成功, 就回调此方法, 在主线程运行

        @Override

        public View onCreateSuccessView()

        {

                /**

                 * 第一步：创建ListView

                 */

                ListView lv = new ListView(UIUtils.getContext());

                lv.setAdapter(new HomeAdapter(data));

                return lv;

 

        }

 

        // 运行在子线程,可以直接执行耗时网络操作

        @Override

        public ResultState onLoad()

        {

                /**

                 * 第二步：获取lv的数据

                 */

                HomeProtocol protocol = new HomeProtocol();

                data = protocol.getData(0);// 加载第一页数据，index为0

 

                return check(data);

                // 校验数据并返回.check方法见BaseFragment,如下：

                /*

                 * // 对网络返回数据的合法性进行校验

        public ResultState check(Object obj)

        {

                if (obj != null)

                {

                        if (obj instanceof ArrayList)

                        {// 判断是否是集合

 

                                ArrayList list = (ArrayList) obj;

 

                                if (list.isEmpty())

                                {

                                        return ResultState.STATE_EMPTY;

                                } else

                                {

                                        return ResultState.STATE_SUCCESS;

                                }

                        }

                }

 

                return ResultState.STATE_ERROR;

        }

                 * */

 

        }

 

        /**

         * 第三步：创建ListView的Adapter

         */

        class HomeAdapter extends MyBaseAdapter<AppInfo>

        {

 

                public HomeAdapter(ArrayList<AppInfo> data)

                {

                        super(data);

                }

 

                @Override

                public BaseHolder<AppInfo> getHolder()

                {

                        return new HomeHolder();

                }

 

                // 此方法在子线程调用

                @Override

                public ArrayList<AppInfo> onLoadMore()

                {

 

                        /**

                         * 第四步：实现加载更多数据的功能

                         */

                        HomeProtocol protocol = new HomeProtocol();

 

                        // 20, 40, 60....

                        // 下一页数据的位置等于当前集合的大小，当前集合加载了多少条，后面的分页就从这个值之后继续加载

                        ArrayList<AppInfo> moreData = protocol.getData(getListSize());
