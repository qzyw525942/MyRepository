异步加载框架如下：

public abstract class MyAsycnTaks

{

         private Handler handler = new Handler()

         {

                   public void handleMessage(android.os.Message msg)

                   {

                            postTask();

                   };

         };

 

         /**

          * 在子线程之前执行的方法

          */

         public abstract void preTask();

 

         /**

          * 在子线程之中执行的方法

          */

         public abstract void doinBack();

 

         /**

          * 在子线程之后执行的方法

          */

         public abstract void postTask();

 

         /**

          * 执行

          */

         public void execute()

         {

                   preTask();

                   new Thread()

                   {

                            public void run()

                            {

                                     doinBack();

                                     handler.sendEmptyMessage(0);

                            };

                   }.start();

         }

}

第二集  ListView的缓存复用

 

如果在不复用缓存的情况下，一直往下拉ListView，当系统回收上方的条目和系统创建下方的条目跟不上时，即回收内存没有增加内存快时，就会出现out of memory的情况。下面来实现ListView的缓存复用，避免这种问题。

 

之前的ListView条目复用方式如下：

。。。。

EditText et3= (EditText) view.findViewById(R.id.tv);

 

。。。。

 

分析上面的程序可见，每创建一个ListView条目，都会调用getView方法，仅仅判断view能不能复用，然后都会去掉用findViewById方法，如果要创建一百个条目，就要执行一百遍findViewById，所以上面这种方式仅仅复用的是view控件，而没有考虑到findViewById，实际上后者也是十分消耗内存的。findViewById寻找控件的方式和遍历二叉树的方式是一样的，在布局文件中从上到下依次去寻找，所以很耗内存。

 

下面来看看完整的缓存复用方式，即考虑了view和findViewById复用的方式。

第四集  ListView的分批(分页)加载

 

ListView一次性加载的数据太多，就会出现卡顿等现象。所以采用分批加载的方式来加载，每次加载几条。分批加载又叫做分页加载。

 

在BlackNumDao中新增如下查询部分数据的方法：

/**

          * 查询部分数据:查询20条数据

          *

          * @param MaxNum

          *            查询的总个数

          *

          * @param startindex

          *            查询的起始位置

          *

          * @return

          */

         public List<BlackNumInfo> getPartBlackNum(int MaxNum, int startindex)

         {

                   List<BlackNumInfo> list = new ArrayList<BlackNumInfo>();

 

                   // 1.获取数据库

                   SQLiteDatabase database = blackNumOpenHlper.getReadableDatabase();

 

                   // 2.查询操作

                   Cursor cursor = database.rawQuery("select blacknum,mode from info order by _id desc limit ? offset ?",

                                     new String[]

                                     { MaxNum + "", startindex + "" });

 

                   // 以下两种方式，效果一样

 

                   // select blacknum from info order by _id desc limit 20 offset 20

                   // limit : 查询的总个数 offset :查询的起始位置

 

                   // select blacknum from info order by _id desc limit 0,20

                   // 前面:查询的起始位置 后面:查询的总个数

 

                   // 3.解析cursor

                   while (cursor.moveToNext())

                   {

                            // 获取查询出来的数据]

                            String blacknum = cursor.getString(0);

                            int mode = cursor.getInt(1);

 

                            BlackNumInfo blackNumInfo = new BlackNumInfo(blacknum, mode);

 

                            list.add(blackNumInfo);

                   }

 

                   // 4.关闭数据库

                   cursor.close();

                   database.close();

 

                   return list;

         }

                   // 方法之前都有@hide，表示这些方法被隐藏了。再仔细观察可见，有的方法实际上又去调用了

                   // ITelephony这个类里面的方法，此时TelephonyManagener相当于中间者。所以对于这些在

                   // import

                   // android.telephony.NeighboringCellInfo;还是会报错，再搜索NeighboringCellInfo.aidl

                   // 同样的复制其包名--->新建同名包---->复制NeighboringCellInfo.aidl文件到这个包下面，

                   // ITelephony.Stub.asInterface(ServiceManager.getService(Context.TELEPHONY_SERVICE))中就只有

                   // 通过反射获取：

                   //                       ITelephony.Stub.asInterface(ServiceManager.getService(Context.TELEPHONY_SERVICE))

                   // 中的参数ServiceManager.getService(Context.TELEPHONY_SERVICE)

                   try

                   {

                            // 1.通过类加载器加载相应类的class文件：两种方式

                            // Class<?> forName = Class.forName("android.os.ServiceManager");

                            Class<?> loadClass =

BlackNumService.class.getClassLoader().loadClass("android.os.ServiceManager");

 

                            // 2.获取类中相应的方法

                            // name:方法名

                            // parameterTypes:方法的参数类型

                            // receiver : 类的实例，此处为空

                            // args : 具体的方法参数

                            // 挂断电话:增加一个权限：CALL_PHONE

                            iTelephony.endCall();

 

                   } catch (Exception e)

                   }

         {

                   PackageManager pm = context.getPackageManager();

 

                   // 获取系统中安装的所有软件信息:参数设为0

                   List<PackageInfo> app = pm.getInstalledPackages(0);

                   for (PackageInfo packageInfo : app)

                   {

                            // 获取包名

                            String packageName = packageInfo.packageName;

 

                            // 获取版本号

                            String versionName = packageInfo.versionName;

 

                            // 获取application，以便获得程序图标和名称等

                            ApplicationInfo applicationInfo = packageInfo.applicationInfo;

                            // 获取应用程序的图标

                            Drawable icon = applicationInfo.loadIcon(pm);

                            // 获取应用程序的名称

                            String name = applicationInfo.loadLabel(pm).toString();

 

                            // 是否是用户程序

                            // 获取应用程序中相关信息,是否是系统程序和是否安装到SD卡

                            boolean isUser;

                            int flags = applicationInfo.flags;

                            if ((applicationInfo.FLAG_SYSTEM & flags) == applicationInfo.FLAG_SYSTEM)

                            {

//这里的判断条件原理如下：

00000001

                                     // 系统程序

                                     isUser = false;

 

                            } else

                            {

                                     // 用户程序

                                     isUser = true;

 

                            }

 

                            // 是否安装到SD卡

                            boolean isSD;

                            if ((applicationInfo.FLAG_EXTERNAL_STORAGE & flags) ==                                                                   applicationInfo.FLAG_EXTERNAL_STORAGE)

//此处原理与上类似
