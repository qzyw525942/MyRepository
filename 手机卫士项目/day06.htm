第五集  查询号码归属地

 

手机号码的前三位用于区分运行商的类型，往后的四位用于区分归属地的类型。所以前七位就可以判断是哪里的号码属于哪个运营商。

 

查询号码归属地的本质就是查询数据库的操作：手机卫士项目的第一天的上课资料中有一个小米数据库，里面总结了全国电话号码的前七位和归属地的一一对应：

data1和data2两个表通过外键将号码和归属地关联起来：data2的id就是data1的outkey

查询时，先查询data1的外键字段outkey；再根据此外键字段查询data2的location字段:

select outkey from data1 where id=1300000;

select location from data2 where id=272;

两条语句综合起来：

select location from data2 where id=(select outkey from data1 where id=1300000);

这样就能通过号码直接查询归属地了。

 

把这个address.db数据库复制到项目的asserts目录下，不能放在注意res目录下，因为后者的文件是要被编译的，而数据库，音频，视频等无需编译，所以直接放在asserts中。而且程序是无法直接打开asserts中的数据库的，只能打开系统自带的SQLite数据库，所以在项目初始化的时候，就应该把这个address.db复制到SQLite数据库中去。

 

项目初始化是在SplashActivity中进行的，所以在SplashActivity中进行数据库拷贝。

/**

          * 拷贝数据库

          */

         private void copyDb()

         {

 

                   // getCacheDir : 获取缓存的路径

                   // getFilesDir : 获取文件的路径

                   File file = new File(getFilesDir(), "address.db");

                   // 判断文件是否存在：如果不存在才去复制数据库，否则就不复制数据库

                   if (!file.exists())

                   {

                            // 从assets目录中将数据库读取出来

 

                            // 1.获取assets的管理者

                            AssetManager am = getAssets();

                            InputStream in = null;

                            FileOutputStream out = null;

 

                            try

                            {

                                     // 2.读取数据库

                                     in = am.open("address.db");

                                     // 写入流

                                     out = new FileOutputStream(file);

 

                                     // 3.读写操作

                                     // 设置缓冲区

                                     byte[] b = new byte[1024];

                                     int len = 0;

                                     while ((len = in.read(b)) != -1)

                                     {

                                               out.write(b, 0, len);

                                     }

 

                            } catch (IOException e)

                            {

 

                                     e.printStackTrace();

 

                            } finally

                            {

                                      // in.close();

                                     // out.close();

 

                                     // 使用XUtils提供的工具类关闭流

                                     IOUtils.closeQuietly(in);

                                     IOUtils.closeQuietly(out);

                            }

                   }

         }

AddressDao如下：

public class AddressDao

{

         /**

          * 打开数据库,查询号码归属地

          *

          * @param num

          *            号码

          * @param context

          *            上下文

          *

          * @return 归属地

          */

         public static String queryAddress(String num, Context context)

         {

                   String location = "";

 

                   // 1.获取数据库的路径

                   File file = new File(context.getFilesDir(), "address.db");

 

                   // 2.打开数据库

                   // getAbsolutePath : 获取文件的绝对路径

                   // factory : 游标工厂

                   // flags:因为只需要查询，所以设置为只读

                   SQLiteDatabase db = SQLiteDatabase.openDatabase(file.getAbsolutePath(), null, SQLiteDatabase.OPEN_READONLY);

 

                   // 3.查询号码归属地

 

                   // 使用正则表达式进行判断输入的号码是否符合电话号码规则

 

                   // ^1[34578]\d{9}$

                   // ^表示开始

                   // 1表示第一位为1

                   // [34578]表示第二位为34578中的任何一位

                   // \d表示0~9任何一位

                   // {9}表示9位

                   // $表示结束

 

                   // 如果要匹配身份证： ^[0-9]{17}[0-9x]$

                   // [0-9]{17}：前十七位都是0~9

                   // [0-9x]最后一位：0~9或x

                   // 前六位:出生地

                   // 往后8位:出生年月日

                   // 剩下4位:前两位:出生编号

                   // 剩下2位:前一位:性别 奇数男 偶数女 最后一位:前17位的校验 x

 

                   if (num.matches("^1[34578]\\d{9}$"))

                   {// 多用一个\，表示转义

 

                            // 因为这里涉及到两个表的联合查询，这是没办法通过query方法进行的，所以直接使用sql语句查询的方式

                            // sql : sql语句

                            // selectionArgs :查询条件的参数

                            // substring : 包含头不包含尾

                            Cursor cursor = db.rawQuery("select location from data2 where id=(select outkey from data1 where id=?)",

                                               new String[]

                                               { num.substring(0, 7) });

 

                            // 4.解析cursor

                            // 因为每个号码对应一个号码归属地,所以查询出来的是一个号码归属地,只有一条数据，没有必要用whlie,用if就可以了

                            if (cursor.moveToNext())

                            {

                                     location = cursor.getString(0);

                            }

 

                            // 5.关闭数据库

                            cursor.close();

 

                   } else

                   {

                            // 对特殊电话做出来

                            switch (num.length())

                            {

 

                            case 3:// 110 120 119 911

                                     location = "特殊电话";

                                     break;

 

                            case 4:// 5554 5556

                                     location = "虚拟电话";

                                     break;

 

                            case 5:// 10086 10010 10000

                                     location = "客服电话";

                                     break;

 

                            // 6位、9位的号码不常见，所以无需判断

                            // 7位、8位的是座机号

                            case 7:// 座机,本地电话

                            case 8:

                                     location = "本地电话";

                                     break;

 

                            default:

                                     // 010 1234567 10位

                                     // 010 12345678 11位

                                     // 0372 12345678 12位

 

                                     // 长途电话

                                     if (num.length() >= 10 && num.startsWith("0"))

                                     {

                                               // 根据区号查询号码归属：区号有三位和四位两种情况

 

                                               // 1.获取号码的区号

                                               // 3位,4位

 

                                               // 3位

                                               String result = num.substring(1, 3);// 010去掉前面的0后为10

 

                                               // 2.根据区号查询号码归属地

                                               Cursor cursor = db.rawQuery("select location from data2 where area=?",                                               new String[]

                                               { result });

 

                                               // 3.解析cursor

                                               if (cursor.moveToNext())

                                               {

                                                        location = cursor.getString(0);

 

                                                        // 截取数据：

                                                        location = location.substring(0, location.length() - 2);

 

                                                        cursor.close();

 

                                               } else

                                               {

                                                        // 3位没有查询到,直接查询4位

 

                                                        // 获取4位的区号

                                                        result = num.substring(1, 4);// 0372去掉前面的0后为372

 

                                                        cursor = db.rawQuery("select location from data2 where area=?",                                                    new String[]

                                                        { result });

 

                                                        if (cursor.moveToNext())

                                                        {

                                                                 location = cursor.getString(0);

 

                                                                 location = location.substring(0, location.length() - 2);

 

                                                                 cursor.close();

                                                        }

                                               }

                                     }

                                     break;

                            }

                   }

 

                   db.close();

 

                   return location;

         }

}
