第一集  概述

 

在某些控件中，src属性指的是前景，background属性指的是背景。

 

android系统自带的控件无法满足实际开发中的需求，这时就需要自定义控件。

 

自定义组合控件分为如下几类：

1、组合已有的控件实现

--优酷菜单案例

--轮播图广告案例

--下拉选择框案例

 

2、继承已有的控件实现（扩展已有的功能）

--下拉刷新功能ListView案例

--侧滑菜单案例

 

 

3、完全自定义（继承View，ViewGroup。前者是Button，TextView之类的父类，后者是LinearLayout之类的父类）

--自定义开关案例

public class MainActivity extends Activity implements OnClickListener

{

 

         // 三层菜单

         private RelativeLayout rl_level1;

         private RelativeLayout rl_level2;

         private RelativeLayout rl_level3;

 

         // 三层菜单默认都显示

         boolean isLevel1Display = true;

         boolean isLevel2Display = true;

         boolean isLevel3Display = true;

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                   super.onCreate(savedInstanceState);

                   setContentView(R.layout.activity_main);

 

                   // 初始化控件

                   initViews();

         }

 

         private void initViews()

         {

                   // 添加点击事件

                   findViewById(R.id.ib_home).setOnClickListener(this);

                   findViewById(R.id.ib_menu).setOnClickListener(this);

 

                   // 找到三层菜单

                   rl_level1 = (RelativeLayout) findViewById(R.id.rl_level1);

                   rl_level2 = (RelativeLayout) findViewById(R.id.rl_level2);

                   rl_level3 = (RelativeLayout) findViewById(R.id.rl_level3);

         }

 

         @Override

         public void onClick(View v)

         {

                   if (AnimationUtils.runningAnimationCount > 0)// AnimationUtils见后

                   {

                            // 如果二三级菜单的动画正在执行, 就取消Home健和Menu健的点击事件

                            // 一定要这样判断，否则会在动画执行期间点击这两个按钮出现问题

                            return;

                   }

 

                   switch (v.getId())

                   {

 

                   // 一级菜单Home健的点击事件（控制二级菜单和三级菜单的显示与否）

                   case R.id.ib_home:

 

                            if (isLevel2Display)// 如果二级菜单显示着

                            {

                                     long delay = 0;

 

                                     // 如果当前三级菜单已经显示, 先转出去

                                     if (isLevel3Display)

                                     {

                                               AnimationUtils.rotateOutAnim(rl_level3, 0);

 

                                               // 旋转完后就不显示了，所以三级菜单置反

                                               isLevel3Display = false;

 

// 第二级菜单应该在三级菜单出去以后延时一段时间才出去，所以//设置延时时间。如果三级菜单已经出去了，

                                               // 则就不执行这里的if语句，delay依然为0，二级菜单就不延时了

                                               delay += 200;

                                     }

 

                                     // 如果当前二级菜单已经显示, 转出去

                                     AnimationUtils.rotateOutAnim(rl_level2, delay);

 

                            } else

                            {

                                     // 如果当前二级菜单没有显示, 转出来

                                     AnimationUtils.rotateInAnim(rl_level2, 0);

 

                            }

 

                            // 旋转完后就不显示了，所以二级菜单置反

                            isLevel2Display = !isLevel2Display;

 

                            break;

 

                   // 二级菜单中间的Menu健的点击事件(控制三级菜单显示与否)

                   case R.id.ib_menu:

 

                            if (isLevel3Display)

                            {

                                     // 如果当前三级菜单已经显示, 转出去

                                     AnimationUtils.rotateOutAnim(rl_level3, 0);

 

                            } else

                            {

                                     // 如果当前三级菜单没有显示, 转出来

                                     AnimationUtils.rotateInAnim(rl_level3, 0);

 

                            }

 

                            // 置反

                            isLevel3Display = !isLevel3Display;

 

                            break;

 

                   default:

                            break;

                   }

         }

 

         // 如果点击了手机自带的菜单按钮，就隐藏或者显示三个菜单

         @Override

         public boolean onKeyDown(int keyCode, KeyEvent event)

         {

                   // keyCode 事件码

                   System.out.println("onKeyDown: " + keyCode);

 

                   if (keyCode == KeyEvent.KEYCODE_MENU)

                   {

 

                            if (AnimationUtils.runningAnimationCount > 0)

                            {

                                     // 如果当前有动画正在执行, 就取消当前手机菜单按钮的点击事件

                                     return true;

                            }

 

                            // 如果按下的是菜单按钮：就分别从最外层到最内层隐藏三个菜单

                            if (isLevel1Display)

                            {

                                     long delay = 0;

 

                                     // 隐藏三级菜单

                                     if (isLevel3Display)

                                     {

                                               AnimationUtils.rotateOutAnim(rl_level3, 0);

 

                                               isLevel3Display = false;

 

                                               delay += 200;// 此时delay=200

                                     }

 

                                     // 隐藏二级菜单

                                     if (isLevel2Display)

                                     {

                                               AnimationUtils.rotateOutAnim(rl_level2, delay);

 

                                               isLevel2Display = false;

 

                                               delay += 200;// 此时delay=400

                                     }

 

                                     // 隐藏一级菜单

                                     AnimationUtils.rotateOutAnim(rl_level1, delay);

 

                            } else

                            {

                                     // 从最里层到最外层顺次转进来

                                     AnimationUtils.rotateInAnim(rl_level1, 0);

 

                                     AnimationUtils.rotateInAnim(rl_level2, 200);

 

                                     AnimationUtils.rotateInAnim(rl_level3, 400);

 

                                     isLevel3Display = true;

 

                                     isLevel2Display = true;

                            }

 

                            isLevel1Display = !isLevel1Display;

 

                            return true;// 一定要返回true，表示消费了当前事件。

                   }

 

                   return super.onKeyDown(keyCode, event);

         }

 

}

 

AnimationUtils如下：

public class AnimationUtils

{

 

         // 正在运行的动画个数

         public static int runningAnimationCount = 0;

 

         /**

          * 旋转出去的动画

          *

          * @param layout

          *            旋转控件

          *

          * @param delay

          *            延时时间

          */

         public static void rotateOutAnim(RelativeLayout layout, long delay)

         {

                   int childCount = layout.getChildCount();

 

                   // 如果控件已经旋转出去隐藏了. 则找到所有的子View, 禁用它们

                   // 因为这里使用的旋转动画属于补间动画，补间动画在实现效果后，其实原图像

//还在原来的位置，

                   // 点击原来的位置还能触发相应的控件的事件

                   for (int i = 0; i < childCount; i++)

                   {

                            layout.getChildAt(i).setEnabled(false);

                   }

 

                   RotateAnimation ra = new RotateAnimation(0f, -180f, // 开始, 结束的角度,

                                                                                                                 // 负号表示逆时针旋转

                                     Animation.RELATIVE_TO_SELF, 0.5f, // 相对的x坐标点(指定旋转中心的x值)

                                     Animation.RELATIVE_TO_SELF, 1.0f); // 相对的y坐标点(指定旋转中心的y

值)

 

 

                   ra.setDuration(500);// 设定旋转时长

                   ra.setFillAfter(true); // 设置动画停留在结束位置，不然控件会返回原来的位置

                   ra.setStartOffset(delay); // 设置动画延时时间

                   ra.setAnimationListener(new MyAnimationListener()); // 添加动画监听

 

                   layout.startAnimation(ra);

         }

 

         /**

          * 旋转进来的动画

          *

          * @param layout

          *            旋转控件

          *

          * @param delay

          *            延时时间

          */

         public static void rotateInAnim(RelativeLayout layout, long delay)

         {

 

                   int childCount = layout.getChildCount();

 

                   // 如果控件已经旋转进来. 则找到所有的子View, 启用它们

                   for (int i = 0; i < childCount; i++)

                   {

                            layout.getChildAt(i).setEnabled(true);

                   }

 

                   RotateAnimation ra = new RotateAnimation(-180f, 0f, // 开始, 结束的角度,

                                                                                                                   // 顺时针返回初始位置

                                     Animation.RELATIVE_TO_SELF, 0.5f, // 相对的x坐标点(指定旋转中心x值)

                                     Animation.RELATIVE_TO_SELF, 1.0f); // 相对的y坐标点(指定旋转中心y值)

 

                   ra.setDuration(500);

                   ra.setFillAfter(true);

                   ra.setStartOffset(delay); // 设置延时

                   ra.setAnimationListener(new MyAnimationListener()); // 添加动画监听

 

                   layout.startAnimation(ra);

         }

 

         /**

          * 动画监听的类

          */

         static class MyAnimationListener implements AnimationListener

         {

 

                   @Override

                   public void onAnimationStart(Animation animation)

                   {

                            runningAnimationCount++;// 动画开始的时候此变量增加

                   }

 

                   @Override

                   public void onAnimationEnd(Animation animation)

                   {

                            runningAnimationCount--;// 动画开始的时候此变量减少

                   }

 

                   @Override

                   public void onAnimationRepeat(Animation animation)

                   {

                   }

 

         }

 

}

                                               {

                                               } catch (InterruptedException e)

                                               {

                                               }

 

                                               runOnUiThread(new Runnable()

                                               {

                                                                 System.out.println("设置当前位置: " + viewPager.getCurrentItem());

/**

 * 自定义开关

 *

 * Android的界面绘制流程

 * 测量                     摆放                绘制

 * measure      ->      layout       ->      draw

 *     |                      |                     |

 * onMeasure -> onLayout -> onDraw 重写这些方法, 实现自定义控件

 *

 * 这些方法都在onResume()之后执行

 *

 * 继承自View的自定义控件没有子控件，所以没有onLayout，只有：

 * onMeasure() (在这个方法里指定自己的宽高) -> onDraw() (绘制自己的内容)

 *

 * 继承自ViewGroup有子控件，所以有onLayout：

 * onMeasure() (指定自己的宽高, 所有子View的宽高)-> onLayout() (摆放所有子View) -> onDraw() (绘制内容)

 */

 

public class ToggleView extends View {

 

         private Bitmap switchBackgroupBitmap; // 开关背景图片

         private Bitmap slideButtonBitmap; // 开关滑块图片

        

         private Paint paint; // 画笔

        

         private boolean mSwitchState = false; // 开关状态, 默认false

         private float currentX;//滑块当前x位置

 

         boolean isTouchMode = false;//用户是否触摸

         private OnSwitchStateUpdateListener ssul;//把当前开关状态传递出去的接口

        

         /**

          * 用于代码中创建自定义控件

          */

         public ToggleView(Context context) {

                  

                   super(context);

                  

                   init();

                  

         }

 

         /**

          * 在xml里使用, 可指定自定义属性

          */

         public ToggleView(Context context, AttributeSet attrs) {

                  

                   super(context, attrs);

                  

                   init();

                  

                   // 获取配置的自定义属性：自定义属性见后

                   // 名称空间：                 

                   String namespace = "http://schemas.android.com/apk/res/com.itheima74.toggleview";

                  

                   //最后一个参数是默认值

                   int switchBackgroundResource = attrs.getAttributeResourceValue(namespace , "switch_background", -1);

                   int slideButtonResource = attrs.getAttributeResourceValue(namespace , "slide_button", -1);

 

mSwitchState = attrs.getAttributeBooleanValue(namespace, "switch_state", false);

 

                  

                   //下面这两个方法见后          

                   setSwitchBackgroundResource(switchBackgroundResource);

                   setSlideButtonResource(slideButtonResource);

                  

                           

         }

 

         /**

          * 用于在xml里使用, 可指定自定义属性, 如果指定了样式, 则走此构造函数

          */

         public ToggleView(Context context, AttributeSet attrs, int defStyle) {

                  

                   super(context, attrs, defStyle);

                  

                   init();

        

         }

        

         /**

          * 初始化方法

          */

         private void init() {

                  

                   //创建画笔              

                   paint = new Paint();

                  

         }

 

        

         /**

          * 测量方法。

          */

         @Override

         protected void onMeasure(int widthMeasureSpec, int heightMeasureSpec) {

                  

                   //switchBackgroupBitmap是在onCreate中创建的，而onMeasure在onResume后执行，

                   //onResume在onCreate之后

                   //所以这里的switchBackgroupBitmap肯定不为空             

                   setMeasuredDimension(switchBackgroupBitmap.getWidth(), switchBackgroupBitmap.getHeight());

        

         }

 

         /**

          * 绘制方法

          *

          *  Canvas 画布, 画板. 在其上绘制的内容都会显示到界面上.

          */

         @Override

         protected void onDraw(Canvas canvas) {

                  

                   // 1. 绘制背景：

                   //第一个参数表示所要绘制的内容     

                   //float left, float top两个参数分别表示

                   //所绘制的内容的左上角相对于自定义控件左上角的偏移量

                   //第四个参数表示画笔          

                   canvas.drawBitmap(switchBackgroupBitmap, 0, 0, paint);

                  

                   // 2. 绘制滑块

                   if(isTouchMode){

                            // 根据当前用户触摸到的位置画滑块

                           

                            // 限定滑块的范围       

                            // 让滑块向左移动自身一半大小的位置

                            float slideLeft = currentX - slideButtonBitmap.getWidth() / 2.0f;

                           

                            int right = switchBackgroupBitmap.getWidth() - slideButtonBitmap.getWidth();

                           

                            // 限定滑块范围

                            if(slideLeft < 0){

                                    

                                     left = 0; // 左边范围

                                    

                            }else if (slideLeft > right) {

                                    

                                     left = right; // 右边范围

                                    

                            }

                            //以上位置的关系如下：

                           

                            canvas.drawBitmap(slideButtonBitmap, slideLeft, 0, paint);

                           

                   }else {

                           

                            // 根据开关状态boolean值, 直接设置图片位置

                            if(mSwitchState){// 开

                                    

                                     int newLeft = switchBackgroupBitmap.getWidth() - slideButtonBitmap.getWidth();

                                    

                                     canvas.drawBitmap(slideButtonBitmap, newLeft, 0, paint);

                           

                            }else {// 关

                                    

                                     canvas.drawBitmap(slideButtonBitmap, 0, 0, paint);

                           

                            }

                   }

                  

         }

        

        

        

         // 重写触摸事件, 响应用户的触摸.

         @Override

         public boolean onTouchEvent(MotionEvent event) {

                  

                   switch (event.getAction()) {

                  

                   case MotionEvent.ACTION_DOWN:

                           

                            isTouchMode = true;

                  

                            System.out.println("event: ACTION_DOWN: " + event.getX());

                  

                            currentX = event.getX();

                  

                            break;

                  

                   case MotionEvent.ACTION_MOVE:

                  

                            System.out.println("event: ACTION_MOVE: " + event.getX());

                  

                            currentX = event.getX();

                  

                            break;

                           

                   case MotionEvent.ACTION_UP:

                  

                            isTouchMode = false;

                  

                            System.out.println("event: ACTION_UP: " + event.getX());

                  

                            currentX = event.getX();

                           

                            //获取背景的中心位置                   

                            float center = switchBackgroupBitmap.getWidth() / 2.0f;

                           

                            // 根据当前按下的位置, 和背景中心的位置进行比较.

                            boolean state = currentX > center;

                           

                            // 如果开关状态变化了, 通知MainActivity开关状态更新.

                            // 如果没变化state = mSwitchState，就不必传出去                           

                            if(state != mSwitchState && ssul != null){

                                    

                                     // 把最新的boolean值传出去了

                                     ssul.onStateUpdate(state);

                           

                            }

                           

                            mSwitchState = state;

                           

                            break;

 

                   default:

                            break;

                   }

 

                   // 重绘界面

                   // 此方法会使onDraw()被重新调用, 里边的变量会重新生效.界面会更新

                   invalidate();

                  

                   return true; // 消费了用户的触摸事件, 才可以收到其他的事件.

                  

         }

 

         /**

          * 设置背景图

          */

         public void setSwitchBackgroundResource(int switchBackground) {

                  

                   switchBackgroupBitmap = BitmapFactory.decodeResource(getResources(), switchBackground);

         }

        

 

         /**

          * 设置滑块图片

          */

         public void setSlideButtonResource(int slideButton) {

                  

                   slideButtonBitmap = BitmapFactory.decodeResource(getResources(), slideButton);

        

         }

 

         /**

          * 设置开关状态

          */

         public void setSwitchState(boolean mSwitchState) {

                  

                   this.mSwitchState = mSwitchState;

        

         }

        

         /**

          *以下两个方法是把当前开关状态传出去的接口。查看MainActivity调用的地方就能明白了

          */

         public interface OnSwitchStateUpdateListener{

                  

                   void onStateUpdate(boolean state);

                   //state是本类中设置的当前开关状态的值，通过此接口和下面这个方法就能把它传递到MainActivity中去

         }

 

         public void setOnSwitchStateUpdateListener(OnSwitchStateUpdateListener ssul) {

                                    

                   this.ssul = ssul;

        

         }

 

}
