StreamUtils如下：

public class StreamUtils

{

 

         public static String streamToString(InputStream in)

         {

                   String result = "";

 

                   try

                   {

                            // 创建一个字节数组写入流

                            ByteArrayOutputStream out = new ByteArrayOutputStream();

 

                            byte[] buffer = new byte[1024];

                            int length = 0;

                            while ((length = in.read(buffer)) != -1)

                            {

                                     out.write(buffer, 0, length);

                                     out.flush();

                            }

 

                            result = new String(out.toByteArray(), "utf-8");//对服务器返回的信息进行解码

 

                            out.close();

                   } catch (Exception e)

                   {

                            e.printStackTrace();

                   }

 

                   return result;

         }

}

线程越多，下载不一定越快，还与本地网络带宽，服务器资源的带宽有关.

public class MuchThreadDown

{

 

         private static int threadCount = 3;// 开启3个线程

         private static int blockSize = 0;// 每个线程下载的大小，初始为0

 

         private static int runningThreadCount = 0;// 当前运行的线程数

 

         private static String path = "http://10.135.63.52/itheima74/feiq.exe";

         public static void main(String[] args)

url.openConnection();

                            openConnection.setConnectTimeout(10 * 1000);

                            if (code == 200)

         public static class DownloadThread extends Thread

         {

 

                   private int threadId;

                   private int startIndex;

                   private int endIndex;

 

                   private int lastPostion;

 

                   public DownloadThread(int threadId, int startIndex, int endIndex)

                   {

                            this.threadId = threadId;

                            this.startIndex = startIndex;

                            this.endIndex = endIndex;

                   }

 

                   @Override

                   public void run()

                   {

 

                            synchronized (DownloadThread.class)

                            {//加个锁，表示每次只能有一个线程执行此处代码

                                     runningThreadCount = runningThreadCount + 1;//每开启一线程,线程数加1

                            }

 

                            // 第二次请求网络：三个线程分别下载所分配的文件到本地

                            try

                            {

                                     URL url = new URL(path);

                                     HttpURLConnection connnection = (HttpURLConnection)

url.openConnection();

 

                                     connnection.setRequestMethod("GET");

                                     connnection.setConnectTimeout(10 * 1000);

 

                                     System.out.println("理论上下载：  线程：" + threadId + "，开始位置：" +

startIndex + ";结束位置:" + endIndex);

 

                                     //读取上次断点下载结束的位置,本次从这个位置开始继续下载。

 

                                     File file2 = new File(threadId + ".txt");

//上次断点下载结束的位置分别保存在以线程ID命名的文本文件中，有

//几个线程就有几个文本文件

 

                                     if (file2.exists())

                                     {

 

//读取文本文件获取上次断点下载结束的位置

                                               BufferedReader bufferedReader = new BufferedReader(

                                                                 new InputStreamReader(new FileInputStream(file2)));

 

                                               String lastPostion_str = bufferedReader.readLine();

//上次断点下载结束的位置存储的数据只有一行，所以读取一行就行

 

                                               lastPostion = Integer.parseInt(lastPostion_str);

 

                                               //设置分段下载的头信息。 Range:做分段数据请求用的。

                                               connnection.setRequestProperty("Range", "bytes:" + lastPostion + "-" +                                        endIndex);

                                               // bytes:0-500:请求服务器资源中0-500之间的字节信息

 

                                               System.out.println("实际下载：  线程：" + threadId + "，开始位置："                                              + lastPostion + ";结束位置:" + endIndex);

 

                                               bufferedReader.close();

//这个流必须关闭，否则它一直连着存储上次断点下载结束位置的文 //本文件threadID.txt，导致文件全部下载完成时无法删除这些文本文     //件

 

                                     } else

                                     {

 

                                              lastPostion = startIndex;

//如果上次没有断点下载，也就是说本次就是第一次下载，就把这次 //开始下载的位置(lastPostion )指向这个线程所分配的那部分文件的         //开始位置(startIndex)

 

                                               //设置分段下载的头信息。 Range:做分段数据请求用的。

                                               connnection.setRequestProperty("Range", "bytes:" + lastPostion + "-"

+ endIndex);

                                               // bytes:0-500:请求服务器资源中0-500之间的字节信息

 

                                               System.out.println("实际下载：线程：" + threadId + "，开始位置：" +                                               lastPostion + ";结束位置:" + endIndex);

                                     }

 

 

                                     if (connnection.getResponseCode() == 206)

                                     {

// 200：请求全部资源成功，

//206：请求部分资源成功,

//这里是多线程，每个线程下载一部分，所以为请求部分资源成功

 

                                              InputStream is = connnection.getInputStream();

                                               // 请求成功将流写入本地文件中，即已经创建的那个空白文件中

 

                                               RandomAccessFile raf =

new RandomAccessFile(new File(getFileName(path)), "rw");

//通过url获得文件名

 

                                               raf.seek(lastPostion);// 设置每个线程从哪个位置开始下载。

 

                                               // 将流中的数据写入文件

                                              byte[] buffer = new byte[1024];

                                               int length = -1;

 

                                               int total = 0;// 记录本次线程下载的总大小

 

                                               while ((length = is.read(buffer)) != -1)

                                               {

                                                        raf.write(buffer, 0, length);

 

                                                        total = total + length;

 

                                                        // 断点下载：需要把当前线程下载的位置保存到文件中

//方便下次续传的时候继续从这儿开始下载

                                                        int currentThreadPostion = lastPostion + total;

// 计算出当前线程本次下载的位置。当前线程下载结束的位置    //是上一次下载结束的位置加上这一次下载的文件长度之和

 

                                                        //创建随机文件保存当前线程下载的位置，断点续传时再从这                                                     //个位置继续下载

                                                        File file = new File(threadId + ".txt");

                                                        RandomAccessFile accessfile = new RandomAccessFile(file, "rwd");

                                                        accessfile.write(String.valueOf(currentThreadPostion).getBytes());

                                                        accessfile.close();

 

                                               }

                                               // 关闭相关的流信息

                                               is.close();

                                               raf.close();

 

                                               System.out.println("线程：" + threadId + "，下载完毕");

 

                                               //当所有线程下载结束，删除存放下载位置的文件threadID。

                                               synchronized (DownloadThread.class)

                                               {//加个锁，表示每次只能有一个线程执行此处代码

 

                                                        runningThreadCount = runningThreadCount - 1;

// 标志着一个线程下载结束。

 

                                                        if (runningThreadCount == 0)//如果正在运行的线程数为0，表示                                                   //全部文件下载完成，就删除三个缓存下载位置的文件

                                                        {

                                                                 System.out.println("所有线程下载完成");

                                                                 for (int i = 0; i < threadCount; i++)

                                                                 {

                                                                           File file = new File(i + ".txt");

                                                                           file.delete();

                                                                 }

                                                        }

 

                                               }

 

                                     }

 

                            } catch (Exception e)

                            {

                                     e.printStackTrace();

                            }

 

                            super.run();

                   }

 

         }

         public class DownloadThread extends Thread

         {

 

                   private int threadId;

                   private int startIndex;

                   private int endIndex;

 

                   private int lastPostion;

 

                   private int currentThreadTotalProgress;

 

                   public DownloadThread(int threadId, int startIndex, int endIndex)

                   {

                            this.threadId = threadId;

                            this.startIndex = startIndex;

                            this.endIndex = endIndex;

 

                            this.currentThreadTotalProgress = endIndex - startIndex + 1;

                            // 当前线程总进度

                   }

 

                   @Override

                   public void run()

                   {

                            // 获取当前线程对应ProgressBar

                            ProgressBar progressBar = map.get(threadId);

 

                            synchronized (DownloadThread.class)

                            {

 

                                     runningTrheadCount = runningTrheadCount + 1;// 开启一线程，线程数加1

                            }

 

                            // 分段请求网络连接，分段保存文件到本地

                            try

                            {

                                     URL url = new URL(path);

                                     HttpURLConnection openConnection =

(HttpURLConnection) url.openConnection();

 

                                     openConnection.setRequestMethod("GET");

                                     openConnection.setConnectTimeout(10 * 1000);

 

                                     System.out.println("理论上下载：线程：" + threadId + "，开始位置：" +                                 startIndex + ";结束位置:" + endIndex);

 

                                     // 读取上次下载结束的位置,本次从这个位置开始直接下载。

                                     //在安卓中不在保存到文本文件中，而是保存在sharedPreferences中

 

                                     if (SharedUtils.getLastPosition(mContext, threadId) != -1)

                                     {// 不等于-1，说明sharedPreferences中有数据

                                     // 等于-1说明是第一次下载

//SharedUtils类见后

 

                                               lastPostion = SharedUtils.getLastPosition(mContext, threadId);

 

                                               if (lastPostion == endIndex + 1)

                                               // 说明该线程已经下载完成

                                               {

 

                                                        progressBar.setProgress(currentThreadTotalProgress);

                                                        runningTrheadCount = runningTrheadCount - 1;

 

                                               }

 

                                               // 设置分段下载的头信息。 Range:做分段数据请求用的。

                                               openConnection.setRequestProperty("Range", "bytes:" + lastPostion +

"-" + endIndex);// bytes:0-500:请求服务器资源中0-500之间的字节信    息

                                                                                                                                                                                                                                System.out.println("实际下载：  线程：" + threadId + "，开始位置："                                              + lastPostion + ";结束位置:" + endIndex);

                                    

                                     } else

                                     {

 

                                               lastPostion = startIndex;

                                               // 设置分段下载的头信息。 Range:做分段数据请求用的。

                                               openConnection.setRequestProperty("Range", "bytes:" + lastPostion +

"-" + endIndex);// bytes:0-500:请求服务器资源中0-500之间的字节信    息

                                                                                                                                                                                                                               

                                               System.out.println("实际下载：线程：" + threadId + "，开始位置："                                         + lastPostion + ";结束位置:" + endIndex);

 

                                     }

 

                                     if (openConnection.getResponseCode() == 206)

                                     {// 200：请求全部资源成功，

// 206代表部分资源请求成功

 

                                               InputStream inputStream = openConnection.getInputStream();

                                               // 请求成功将流写入本地文件中，已经创建的占位那个文件中

 

                                               RandomAccessFile randomAccessFile =

new RandomAccessFile(new File(getFileName(path)), "rw");

                                               randomAccessFile.seek(lastPostion);//设置从哪个位置开始写。

 

                                               // 将流中的数据写入文件

                                               byte[] buffer = new byte[1024 * 100];

                                               int length = -1;

                                               int total = 0;// 记录本次线程下载的总大小

 

                                               while ((length = inputStream.read(buffer)) != -1)

                                               {

                                                        randomAccessFile.write(buffer, 0, length);

 

                                                        total = total + length;

                                                        // 去保存当前线程下载的位置，保存到文件中

                                                        int currentThreadPostion = lastPostion + total;

//计算出当前线程本次下载的位置

                                                       

                                                        //把当前已经下载的位置封装到SharedPreferences中，而不再                                                    //是文件中

 

                                               SharedUtils.setLastPosition(mContext,threadId, currentThreadPostion);

 

                                                        //计算线程下载的进度并设置进度

                                                        int currentprogress = currentThreadPostion - startIndex;

 

                                                        progressBar.setMax(currentThreadTotalProgress);

                                                        // 设置进度条的最大值

 

                                                        progressBar.setProgress(currentprogress);

                                                        // 设置进度条当前进度。子线程中可以更新progressBar

 

                                               }

 

                                               // 关闭相关的流信息

                                               inputStream.close();

                                               randomAccessFile.close();

 

                                               System.out.println("线程：" + threadId + "，下载完毕");

 

                                               // 当所有线程下载结束，删除存放下载位置的文件。

                                               synchronized (DownloadThread.class)

                                               {

                                                        runningTrheadCount = runningTrheadCount - 1;

//标志着一个线程下载结束。

                                                        if (runningTrheadCount == 0)

                                                        {

 

//文件下载完成，弹出吐司

                                                                 runOnUiThread(new Runnable()

                                                                 {

 

                                                                           @Override

                                                                           public void run()

                                                                           {

                                                                                    Toast.makeText(mContext, "下载完毕", 0).show();

                                                                           }

                                                                 });

 

                                                                 System.out.println("所有线程下载完成");

                                                        }

 

                                               }

 

                                     }

 

                            } catch (Exception e)

                            {

                                     e.printStackTrace();

                            }

 

                            super.run();

                   }

 

         }

         }

SharedUtils如下：

public class SharedUtils

{

 

         public static int getLastPosition(Context context, int threadId)

         {

 

                   SharedPreferences sp = PreferenceManager.getDefaultSharedPreferences(context);

                   return sp.getInt("lastPostion" + threadId, -1);

                   //如果sp中没有增加新的数据， 默认值就设置为-1，如果有新建的数据，则这个            //-1就会被替代.依次来判断线程是否为第一次下载。

 

         }

 

         public static void setLastPosition(Context context, int threadId, int position)

         {

                   SharedPreferences sp = PreferenceManager.getDefaultSharedPreferences(context);

                   sp.edit().putInt("lastPostion" + threadId, position).commit();

         }

}
