A、线性布局(LinearLayout)

 

layout目录上右键-->new-->Android Xml File，

 

新建的LinearLayout如下：

 

<?xml version="1.0" encoding="utf-8"?>//xml文件的文档声明

<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"//名称空间

 

android:layout_width="match_parent"

//共有三种取值：match_parent：匹配父窗体；fill_parent：填充父窗体；wrap_content:  //包裹内容

android:layout_height="match_parent"

 

android:orientation="vertical"

//线性布局的方向：vertical：垂直方向；"horizontal"：水平方向

   

android:gravity="center">

//对齐方式，子控件相对于当前控件的对齐方式.

//android:layout_gravity="center":当前控件相对于父控件的对齐方式

//二者是相对的，子控件的layout_gravity也就是父控件的gravity。任意设置其中之一即          //可

 

 

 <TextView

        android:layout_height="wrap_content"

        android:layout_width="wrap_content"

        android:text="你好！"

        android:gravity="center"/>

 

 <EditText

        android:layout_width="wrap_content"

        android:layout_height="wrap_content"

        android:hint="你好啊"//设置EditText初始提示信息

        android:background="#00ff55"/>//设置背景色

 

</LinearLayout>

 

此外：

含有的margin的设置:当前控件相对于四周的间距。

含有的padding的设置:当前控件中的子控件相对于当前控件四周的间距。

二者是相对的，子控件的margin也就是父控件的padding。任意设置其中之一即可

第三集    Logcat的使用

 

在某一个代码中增加System.out.println(“XXX”);然后运行这个代码所在的程序，如果在logcat中显示了这个打印信息，则说明这句输出语句之前的代码都没有问题，这是调试代码的一种方法。

 

如果程序已经在虚拟机上运行起来了，但是Logcat没有显示任何信息，那么在devices窗口中在要显示信息的avd上点击一下，logcat就能显示出日志信息。

 

Logcat显示出的错误日志中，先找Cause by语句，看它是怎么说的；如果没有这一句再看其他错误日志

 

logcat的窗口如下：

其中1处的verbose表示任何信息都会输出，debug输出调试的信息，info输出一般的提示信息，warn输出警告信息，error输出错误信息。

2处可以搜索信息，比如代码中有System.out.println(“XXX”)，就可以直接输入XXX就能找到这个信息。

3处是保存日志的按钮：选中某一行信息，按住shift，再选另一行，则两行之间的信息就被选中了，点击保存按钮，就能把选中的信息保存为txt格式。

4处为新建日志信息过滤器的按钮，如下图所示，完善信息后就能从logcat窗口中过滤出指定的信息。

封装一个工具类LogUtils

public class LogUtils

{

 

         private static boolean enableLog = false;

         //测试中把这儿设置为true，程序就会执行测试代码，

         //设置为false，程序就不会执行测试的代码，并且测试

         //代码不删除也不影响程序执行效率

 

         public static void e(String tag, String msg)

         {

 

                  if (enableLog)

                  {

                           Log.e(tag, msg);

                  }

         }

 

         public static void w(String tag, String msg)

         {

                  if (enableLog)

                  {

                           Log.w(tag, msg);

                  }

         }

 

         public static void i(String tag, String msg)

         {

                  if (enableLog)

                  {

                           Log.i(tag, msg);

                  }

         }

 

         public static void d(String tag, String msg)

         {

                  if (enableLog)

                  {

                           Log.d(tag, msg);

                  }

         }

 

         public static void v(String tag, String msg)

         {

                  if (enableLog)

                  {

                           Log.v(tag, msg);

                  }

         }

}

UserInfoUtil工具类如下：

public class UserInfoUtil

{

 

// 保存用户名密码

         public static boolean saveUserInfo(Context context, String username, String password)

         {

 

                  try

                  {

// 1、拼接用户名密码

                           String userinfo = username + "##" + password;

 

                           //2、通过Context对象获取私有目录                                       

/data/data/com.itheima.login/files。注意这个Context对象                       

//是MainActivity的对象，而不是工具类的对象，所以通过方法的参数传递进    //来。

                           String path = context.getFilesDir().getPath();

 

                           System.out.println("...............:" + path);

 

//3、创建保存信息的文件

                           File file = new File(path, "userinfo.txt");

 

//4、创建文件输出流

                           FileOutputStream fos= new FileOutputStream(file);

 

 

                           fos.write(userinfo.getBytes());// 将用户名密码写入文件

 

                           fos.close();

 

                           return true;

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

 

                  return false;

         }

 

         // 获取用户名密码

         public static Map<String, String> getUserInfo(Context context)

         {

 

                  try

                  {

                           // 1、通过Context对象获取私有目录

                           String path = context.getFilesDir().getPath();

 

//2、获得保存信息的文件

                           File file = new File(path, "userinfo.txt");

 

//3、获得文件输入流

                           FileInputStream fis= new FileInputStream(file);

 

                           BufferedReader br=

new BufferedReader(new InputStreamReader(fis));

//包装流：将字节流转为字符流，以便读取一行，因为保存的时候保存为一行

 

                           // 读取一行中包含用户密码，需要解析

                           String readLine = br.readLine();

 

//4、通过”##”去切割用户名和密码两个信息

                           String[] split = readLine.split("##");

 

                           HashMap<String, String> hashMap = new HashMap<String, String>();

 

                           hashMap.put("username", split[0]);

 

                           hashMap.put("password", split[1]);

 

                           br.close();

 

                           fis.close();

 

                           return hashMap;

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

                  return null;

 

         }

 

}

 

上面的案例通过context.getFilesDir().getPath();方法把用户信息保存在此路径中。这个路径是程序的私有路径，属于系统为每个应用程序指定的保存数据的路径。现在把信息保存在/mnt/sdcard中，这个路径属于外部扩展的存储路径，即SD卡中。

         是一个应用程序的私有目录，只有当前应用程序有权限访问读写，其他应用无权限访问。一些安全性要求比较高的数据存放在该目录，一般用来存放size比较小的数据

 

/sdcard: Enviroment.getExternalStorageDirectory().getPath();

         是一个外部存储目录，只要用户声明了

<uses-permission android:name="android.permission.WRITE_EXTERNAL_STORAGE"/>的一个权限，就可以访问读写sdcard目录；所以一般用来存放一些安全性不高的数据，文件size比较大的数据。

public class UserInfoUtil

{

 

         // 保存用户名密码

         public static boolean saveUserInfo_android(Context context, String username, String password)

         {

 

                  try

                  {

 

                           // 1.通过Context对象创建一个SharedPreference对象

 

                           SharedPreferences  sp

= context.getSharedPreferences("userinfo.txt", Context.MODE_PRIVATE);

      // 两个参数：name:sharedpreference文件的名称

                           // mode:文件的操作模式

//sharedPreferences保存的数据文件在/data/data/包名/shared_prefs中

 

 

                           // 2.通过sharedPreferences对象获取一个Editor对象

                           Editor editor = sp.edit();

 

                           // 3.往Editor中添加数据

                           editor.putString("username", username);

                           editor.putString("password", password);

 

                           // 4.提交Editor对象

                           editor.commit();

 

                           return true;

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

 

                  return false;

         }

 

         // 获取用户名密码

         public static Map<String, String> getUserInfo_android(Context context)

         {

 

                  try

                  {

 

                           // 1.通过Context对象创建一个SharedPreference对象

                           SharedPreferences sp

= context.getSharedPreferences("userinfo.txt", Context.MODE_PRIVATE);

 

                           // 2.通过sp获取存放的数据

                           // key:存放数据时的key

                           // defValue: 默认值,根据业务需求来写,第一次登录时系统还没有任何数据，

                           // 如果选择记住用户名和密码,则显示的就是这个参数的值，这里为空，则第                    //一次登陆不显示任何数据

                           String username = sp.getString("username", "");

                           String password = sp.getString("password", "");

 

                           HashMap<String, String> hashMap = new HashMap<String, String>();

                           hashMap.put("username", username);

                           hashMap.put("password", password);

 

                           return hashMap;

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

                  return null;

 

         }
