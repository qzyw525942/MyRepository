AccountProvider如下：

public class AccountProvider extends ContentProvider

{// 创建一个ContentProvider的步骤：

// A、在要提供数据的工程中创建一个类，继承自ContentProvider

// B、实现以下几个方法

// C、在清单文件中如下配置：

// <provider

// android:name="com.itheima.db.AccountProvider"

// android:authorities="com.itheima.provider" >

// </provider>

// 这里配置主机名authorities，其他程序通过这个由主机名构成的URI访问本程序的数据，见下面代码中更进一步的说明

 

         // [1]定一个uri路径匹配器

         private static final UriMatcher URIMatcher = new UriMatcher(UriMatcher.NO_MATCH);

         // 匹配不成功的时候，使用默认的UriMatcher.NO_MATCH

         // 从这里开始直到oncreate之前的代码都可在UriMatcher类中找到实例：鼠标在UriMatcher类上悬停就能看到

 

         private static final int QUERYSUCESS = 0;

         // ctrl+shift+X 小写变大写，ctrl+shift+y大写变小写

         private static final int INSERTSUCESS = 1;

         private static final int UPDATESUCESS = 2;

         private static final int DELETESUCESS = 3;

 

         private MyOpenHelper myOpenHelper;

 

         // [2]创建一个静态代码块 在这个里面添加 uri

         static

         {

 

                  // authority主机名，其和本ContentProvider清单文件里面定义的authorities的值一样

                  // 如下配置好后其他程序就能通过com.itheima.provider/query访问本程序中对应的方法了

                  // code为访问相应方法成功与否的状态码

 

                  URIMatcher.addURI("com.itheima.provider", "query", QUERYSUCESS);

                  URIMatcher.addURI("com.itheima.provider", "insert", INSERTSUCESS);

                  URIMatcher.addURI("com.itheima.provider", "update", UPDATESUCESS);

                  URIMatcher.addURI("com.itheima.provider", "delete", DELETESUCESS);

         }

 

         // 当内容提供者初始化时会执行此方法

         @Override

         public boolean onCreate()

         {

 

                  // [3]初始化 myopenHelpler对象 就可以获取到sqlitedatabases对象 我们就可以操作数据库了

 

                  myOpenHelper = new MyOpenHelper(getContext());

 

                  return false;

         }

 

         // 对外暴露查询的方法

         @Override

         public Cursor query(Uri uri, String[] projection, String selection, String[] selectionArgs, String sortOrder)

         {

                  int code = URIMatcher.match(uri);

 

                  if (code == QUERYSUCESS)

                  {

                           // 外界传入的uri与QUERYSUCESS相同，说明查询方法的路径匹配成功

 

                           SQLiteDatabase db = myOpenHelper.getReadableDatabase();

 

                           // 调用query方法

                           Cursor cursor = db.query("info", projection, selection, selectionArgs, null, null, sortOrder);

 

                           // 发送一条消息，说明数据库被操作了

                           getContext().getContentResolver().notifyChange(uri, null);

 

                           // db.close();

 

                           // 重点：这个cursor不能关

                           return cursor;

 

                  } else

                  {

                           // 说明路径不匹配

                           throw new IllegalArgumentException("uri路径不匹配 请检测路径！");

 

                  }

                  // 写到这里运行此程序，如果发现如下日志：

                  // Pub com.itheima.provider: com.itheima.db.AccountProvider

                  // 就说明内容提供者写的没有问题

 

         }

 

         // 对外暴露增加的方法

         @Override

         public Uri insert(Uri uri, ContentValues values)

         {

 

                  int code = URIMatcher.match(uri);

 

                  if (code == INSERTSUCESS)

                  {

                           // 外界传入的uri与INSERTSUCESS相同，说明增加方法的路径匹配成功

 

                           SQLiteDatabase db = myOpenHelper.getReadableDatabase();

 

                           long insert = db.insert("info", null, values);

                           // 返回值表示插入的行数

 

                           Uri uri2 = Uri.parse("com.hahaheheheihei/" + insert);

                           // 这个uri的值可以任意写。第二个程序看到这个返回的URI就知道插入成功

 

                           if (insert > 0)

                           {

                                    // 发送一条消息 说明说明数据库被操作了

                                    getContext().getContentResolver().notifyChange(uri, null);

                           }

 

                           db.close();// 关闭数据库

 

                           return uri2;

 

                  } else

                  {

                           throw new IllegalArgumentException("uri路径不匹配 请检测路径");

                  }

 

         }

 

         // 对外暴露删除的方法

         @Override

         public int delete(Uri uri, String selection, String[] selectionArgs)

         {

 

                  int code = URIMatcher.match(uri);

 

                  if (code == DELETESUCESS)

                  {

                           // 外界传入的uri与DELETESUCESS相同，说明删除方法的路径匹配成功

 

                           SQLiteDatabase db = myOpenHelper.getReadableDatabase();

 

                           int delete = db.delete("info", selection, selectionArgs);

                           // 返回值代表影响的行数

 

                           if (delete > 0)

                           {

                                    // 发送一条消息 说明说明数据库被操作了

                                    getContext().getContentResolver().notifyChange(uri, null);

 

                           }

 

                           return delete;

 

                  }

 

                  return 0;

         }

 

         // 对外暴露修改的方法

         @Override

         public int update(Uri uri, ContentValues values, String selection, String[] selectionArgs)

         {

                  int code = URIMatcher.match(uri);

 

                  if (code == UPDATESUCESS)

                  {

                           // 外界传入的uri与UPDATESUCESS相同，说明修改方法的路径匹配成功

                           SQLiteDatabase db = myOpenHelper.getWritableDatabase();

 

                           int update = db.update("info", values, selection, selectionArgs);

                           // 返回值代表影响的行数

 

                           if (update > 0)

                           {

                                    // 发送一条消息 说明说明数据库被操作了

                                    getContext().getContentResolver().notifyChange(uri, null);

 

                           }

 

                           return update;

 

                  } else

                  {

                           throw new IllegalArgumentException("uri路径不匹配 请检测路径");

 

                  }

 

         }

 

         @Override

         public String getType(Uri uri)

         {

                  return null;

         }

 

}

                  Uri uri = Uri.parse("content://com.itheima.provider/query");

                  // 外界提供第一个程序的内容提供者提供的URI路径去访问第一个程序暴露的query方法

 

                  // 通过内容解析者获取数据

                  Cursor cursor = getContentResolver().query(uri, new String[]

                  { "name", "money" }, null, null, null);

 

                  if (cursor != null)

                           }

                  }

 

         }

public class MainActivity extends Activity

{

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                  super.onCreate(savedInstanceState);

                  setContentView(R.layout.activity_main);

         }

 

         // 点击按钮查询短信内容 然后把短信内容进行备份

         public void click(View v)

         {

                  try

                  {

                           // [1]获取XmlSerializer的实例

                           XmlSerializer serializer = Xml.newSerializer();

 

                           // [2]设置序列化器参数

                           File file = new File(Environment.getExternalStorageDirectory().getPath(), "smsbackup.xml");

 

                           FileOutputStream fos = new FileOutputStream(file);

 

                           serializer.setOutput(fos, "utf-8");

 

                           // [3]写xml文档开头

                           serializer.startDocument("utf-8", true);

 

                           // [4]写xml的根节点

                           serializer.startTag(null, "smss");

 

                           // [5]构造uri：短信应用程序通过这个URL把其内的数据暴露出来。这个URL                      //是通过查阅资料获得的

                           Uri uri = Uri.parse("content://sms/");

 

                           // [6]由于短信的数据库已经通过内容提供者暴露出来 所以我们直接通过内                         //容解析者查询数据

 

                           Cursor cursor = getContentResolver().query(uri, new String[]

                           { "address", "date", "body" }, null, null, null);

 

                           while (cursor.moveToNext())

                           {

                                    String address = cursor.getString(0);

                                    String date = cursor.getString(1);

                                    String body = cursor.getString(2);

 

                                    // [7]写sms节点

                                    serializer.startTag(null, "sms");

 

                                    // [8]写address节点

                                    serializer.startTag(null, "address");

                                    serializer.text(address);

                                    serializer.endTag(null, "address");

 

                                    // [9]写date节点

                                    serializer.startTag(null, "date");

                                    serializer.text(date);

                                    serializer.endTag(null, "date");

 

                                    // [10]写body节点

                                    serializer.startTag(null, "body");

                                    serializer.text(body);

                                    serializer.endTag(null, "body");

 

                                    serializer.endTag(null, "sms");

 

                           }

 

                           serializer.endTag(null, "smss");

 

                           serializer.endDocument();

 

                           fos.close();

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

 

         }

 

}

第四集   内容观察者

 

原理：

 

public class MainActivity extends Activity

{

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                  super.onCreate(savedInstanceState);

                  setContentView(R.layout.activity_main);

 

                  // [1]注册内容观察者：观察第一集那个增删改查案例中，第一个提供数据的程序              //的数据内容是否发生了改变

                  Uri uri = Uri.parse("content://com.itheima.provider/");

 

                  // false:观察的uri必须是一个确切的 uri：content://com.itheima.provider/query

//true:观察的uri可以不必是一个确切的 uri： content://com.itheima.provider/

                  getContentResolver().registerContentObserver(uri, true, new MyContentObserver(new                   Handler()));

 

         }

 

         // 定义内容观察者

         private class MyContentObserver extends ContentObserver

         {

 

                  public MyContentObserver(Handler handler)

                  {

                           super(handler);

                  }

 

                  // 当我们观察的uri所指定的内容发生改变的时候调用

                  @Override

                  public void onChange(boolean selfChange)

                  {

                           System.out.println("哈哈 数据库被操作了 ");

 

                           super.onChange(selfChange);

                  }

 

         }

 

}

 

并且在第一集那个增删改查案例中，第一个提供数据的程序里相应的增删改查四个方法中都加入如下发送通知的代码：

 

// 向内容观查者发送一条消息，说明数据库被操作了：

getContext().getContentResolver().notifyChange(uri, null);

 

这样当第一集那个增删改查案例中，第二个获取数据的程序对第一个程序进行增删改查操作时，就会触发这里的内容观察者的onChange方法，从而打印出相应信息。
