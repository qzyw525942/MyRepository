第一集  计算机中表示图片的方式

 

在画图软件上固定画布大小为800x400像素，并随意做一幅画，点击另存为：

可见所能保存的图片有bmp，jpeg和png三种。

 

选择bmp格式，如下图所示：

可见具有单色位图，16色位图，256色位图和24位位图四种。

 

把上面画的图像分别保存为这四种格式的图像共四张：

查看这四张图片的大小，可见

单色---->40,062 字节

16色---->160,118 字节

256色---->321,078 字节

24位----->960,054 字节

 

下面来看这些字节数是怎么来的：

图片大小=图片总像素*每个像素的大小

 

图片的总像素上面已经固定为800*400=320000像素；

上面保存的四张图像大小不一样，说明是每个像素的大小不一样。

 

单色位图： 每个像素最多可以表示2种颜色(0表示白，1表示黑)只需要使用长度为1的二               进制位来表示，那么每个像素占1/8byte。所以上面的单色图大小为                                  320000*1/8byte=40000byte。多出来的62个字节用于保存图片的额外信息，比                     如拍摄时间等

16色位图：每个像素最多可以表示16种颜色，0000 - 1111 那么只需要使用长度为4的二                        进制表示 那么每个像素占1/2个byte。所以上面的16色图大小为                                        320000*1/2byte=160000byte。多出来的118个字节用于保存图片的额外信息，                     比如拍摄时间等

256色位图：每个像素最多可以表示256种颜色，0000 0000 - 1111 1111  那么只需要使用                        长度8的二进制位表示，那么每个像素占1byte。所以上面的256色图大小为                         320000*1byte=320000byte。多出来的1,078个字节用于保存图片的额         外信息，                           比如拍摄时间等

24位位图：注意这里叫24位。即通常的RGB颜色。R占一个byte，G占一个byte，B占一              个byte，所以每个像素占3个byte。所以上面的24位位图大小为                  

          320000*3byte=960000byte。多出来的54个字节用于保存图片的额外信息，比如  

          拍摄时间等

 

jpg格式：由bmp格式转化而来。二者的区别：bmp可以放大一百倍还能看清楚每个细节，           jpg只能放大30来倍，再放大就看不清细节了。所以jpg占用的内存远远低于bmp

 

png格式：android中采用此格式。由bmp格式转化而来。原理同jpg的。但是所能放大的          倍数更小

点击上面的加载按钮，把图片显示在下面的ImageView中。

 

public class MainActivity extends Activity

{

 

         private ImageView iv;

         private int screenWidth;

         private int screenHeight;

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                  super.onCreate(savedInstanceState);

                  setContentView(R.layout.activity_main);

 

                  // [1]找到iv 显示加载图片

 

                  iv = (ImageView) findViewById(R.id.iv);

 

                  // [2]获取手机的分辨率获取windowmanager实例

                  WindowManager wm = (WindowManager) getSystemService(WINDOW_SERVICE);

                  screenWidth = wm.getDefaultDisplay().getWidth();

                  screenHeight = wm.getDefaultDisplay().getHeight();

 

                  System.out.println("手机的宽和高:" + screenWidth + "---" + screenHeight);

 

         }

 

         // 点击按钮加载dog.jpg这张图片(2400*3200像素)

         // 图片像素太大，而手机的像素为320*480，显示不了，所以要对图像进行缩放

         public void click(View v)

         {

                  // [2]把dog.jpg转换成bitmap

 

                  // 创建bitmap工厂的配置参数

                  BitmapFactory.Options options = new Options();

 

                  // 设置为true，不去真正解析位图，而是返回图片的一些信息(宽和高)

                  options.inJustDecodeBounds = true;

                  BitmapFactory.decodeFile("/mnt/sdcard/dog.jpg", options);

                  // 把一个路径上的jpg图片转化为bmp位图.

 

                  // [3]获取图片的宽和高

                  int imgWidth = options.outWidth;

                  int imgHeight = options.outHeight;

 

                  System.out.println("图片的宽:" + imgWidth + "-----" + imgHeight);

 

                  // [4]计算缩放比

                  int scale = 1; // 我们定义的缩放比

                  int scalex = imgWidth / screenWidth;

                  int scaley = imgHeight / screenHeight;

 

                  // scalex和scaley谁大就用谁。并且缩放比要大于1才有意义

                  if (scalex >= scaley && scalex > scale)

                  {

                           scale = scalex;

                  }

 

                  if (scaley > scalex && scaley > scale)

                  {

                           scale = scaley;

                  }

 

                  System.out.println("缩放比为:" + scale);

 

                  // [5]按照缩放比显示图片

                  options.inSampleSize = scale;

 

                  // [6]开始真正的解析位图

                  options.inJustDecodeBounds = false;

                  Bitmap bitmap = BitmapFactory.decodeFile("/mnt/sdcard/dog.jpg", options);

 

                  // [7]把bitmap显示到控件上

                  iv.setImageBitmap(bitmap);

 

         }

 

}

public class MainActivity extends Activity

                  // [3]拷贝原图

 

                  // [3.1]创建模板

                  Bitmap copybitmap = Bitmap.createBitmap(srcBitmap.getWidth(), srcBitmap.getHeight(), srcBitmap.getConfig());

                  Canvas canvas = new Canvas(copybitmap);

 

                  // [3.3]创建一个画笔

                  Paint paint = new Paint();

 

                  // [3.4]开始作画 参考原图srcBitmap去画

                  // Matrix：矩阵

                  canvas.drawBitmap(srcBitmap, new Matrix(), paint);

 

                  // 复制的图片可以进行编辑了

                  for (int i = 0; i < 10; i++)

                  {

                           // [一次修改一个像素]

                           copybitmap.setPixel(20 + i, 30, Color.RED);

                  }

 

                  // [4]把copybitmap显示到iv_copy上

                  iv_copy.setImageBitmap(copybitmap);

 

         }

 

}

第五集  画画板案例

 

public class MainActivity extends Activity

{

 

         private Bitmap srcBitmap;

         private ImageView iv;

         private Bitmap copyBitmap;

         private Canvas canvas;

         private Paint paint;

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                  super.onCreate(savedInstanceState);

                  setContentView(R.layout.activity_main);

 

                  // [1]找到imageview 显示我们画的内容

                  iv = (ImageView) findViewById(R.id.iv);

 

                  // [2]把bg转换成bitmap

                  srcBitmap = BitmapFactory.decodeResource(getResources(), R.drawable.bg);

                  // bg.png是纯色图片，背景色任意

 

                  // [2.1]创建 bg.png的模板，刚创建的模板为空白的

                  copyBitmap = Bitmap.createBitmap(srcBitmap.getWidth(), srcBitmap.getHeight(), srcBitmap.getConfig());

 

                  // [2.2]在copybitmap模板上创建一个画布

                  canvas = new Canvas(copyBitmap);

 

                  // [2.3]创建一个画笔

                  paint = new Paint();

 

                  // [2.4]开始作画

                  canvas.drawBitmap(srcBitmap, new Matrix(), paint);

 

                  // canvas.drawLine(20, 20, 30, 50, paint);

                  // 画一条线：起点是(20,20)，终点是(30,50)。注意这儿已被注释

 

                  // [3]把copybitmap显示到iv上

                  iv.setImageBitmap(copyBitmap);

 

                  // [4]给iv设置一个触摸事件：触摸iv画画

                  iv.setOnTouchListener(new OnTouchListener()

                  {

 

                           int startX = 0;

                           int startY = 0;

 

                           @Override

                           public boolean onTouch(View v, MotionEvent event)

                           {// 当触摸iv的时候触发此方法

 

                                    // [5]获取手指触摸的事件类型

                                    int action = event.getAction();

 

                                    // [6]具体判断一下是什么事件类型

                                    switch (action)

                                    {

                                    case MotionEvent.ACTION_DOWN: // 按下

 

                                             // [7]获取手指按下坐标

                                             startX = (int) event.getX();

                                             startY = (int) event.getY();

                                             System.out.println("按下:" + startX + "---" + startY);

                                             break;

 

                                    case MotionEvent.ACTION_MOVE:// 移动

 

                                             // [8]获取停止的坐标

                                             int stopX = (int) event.getX();

                                             int stopY = (int) event.getY();

 

                                             System.out.println("移动:" + stopX + "---" + stopY);

 

                                             // [9]画线

                                             canvas.drawLine(startX, startY, stopX, stopY, paint);

 

                                             // [9.1]更新一下起点坐标

                                             startX = stopX;

                                             startY = stopY;

 

                                             // [10]记得更新ui

                                             iv.setImageBitmap(copyBitmap);

 

                                             break;

 

                                    case MotionEvent.ACTION_UP: // 抬起;

 

                                             break;

                                    }

 

                                    return true;

                                    // 必须返回true，这三个事件才能都被执行，如果返回false

                                    // 则只能执行ACTION_DOWN事件

                           }

                  });

 

         }

 

         // 点击按钮让画笔的颜色 变成红色

         public void click1(View v)

         {

                  // 设置画笔颜色

                  paint.setColor(Color.RED);

         }

 

         // 让画笔颜色变粗

         public void click2(View v)

         {

                  // 设置画笔的宽度

                  paint.setStrokeWidth(5);

 

         }

 

         // 保存大作

         public void click3(View v)

         {

                  /**

                   * format 保存图片的格式

                   *

                   * quality 保存照片的质量

                   */

                  try

                  {

                           File file = new File(Environment.getExternalStorageDirectory().getPath(), "dazuo.png");

 

                           FileOutputStream fos = new FileOutputStream(file);

 

                           // 保存图片：

                           // 第一个参数为保存图片的格式，

                           // 第二个参数为保存的质量。范围为0~100

                           // 第三个参数为流

                           copyBitmap.compress(CompressFormat.PNG, 100, fos);

 

                           // 发送一条sd卡挂载上来的广播 欺骗一下系统图库应用 说sd卡被挂载了 你去加载图片吧

 

                           Intent intent = new Intent();

 

                           // 设置action

                           intent.setAction(Intent.ACTION_MEDIA_MOUNTED);

 

                           // 设置data

                           intent.setData(Uri.fromFile(Environment.getExternalStorageDirectory()));

 

                           // 发送无序广播

                           sendBroadcast(intent);

 

                           fos.close(); // 关闭流

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

 

         }

 

}

 

第六集  撕衣服案例

 

界面布局文件如下：

第一个ImageView是呈现撕开后的图片的；第二个ImageView是呈现撕开前的图片的。前者图片在布局文件中指定，后者图片在代码中加载。

 

public class MainActivity extends Activity

{

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                  super.onCreate(savedInstanceState);

                  setContentView(R.layout.activity_main);

 

                  // [1]找到iv 显示我们操作的图片

                  final ImageView iv = (ImageView) findViewById(R.id.iv);

 

                  // [2]把我们要操作的图片转换成bitmap

                  Bitmap srcBitmap = BitmapFactory.decodeResource(getResources(), R.drawable.pre19);

 

                  // [3]创建原图的副本

 

                  // [3.1]创建模板

                  final Bitmap alterBitmap = Bitmap.createBitmap(srcBitmap.getWidth(), srcBitmap.getHeight(),

                                    srcBitmap.getConfig());

 

                  // [3.2]以alterbBitmap bitmap为模板创建一个画布

                  Canvas canvas = new Canvas(alterBitmap);

 

                  // [3.3]创建一个画笔

                  Paint paint = new Paint();

 

                  // [3.4]开始作画

                  canvas.drawBitmap(srcBitmap, new Matrix(), paint);

 

                  // [4]把alterbitmap显示到iv上

                  iv.setImageBitmap(alterBitmap);

 

                  // [5]给iv设置一个触摸事件

                  iv.setOnTouchListener(new OnTouchListener()

                  {

 

                           @Override

                           public boolean onTouch(View v, MotionEvent event)

                           {

                                    // [6]获取触摸事件的类型

                                    int action = event.getAction();

                                    switch (action)

                                    {

 

                                    case MotionEvent.ACTION_MOVE: // 移动

 

                                             for (int i = -7; i < 7; i++)

                                             { // 改变x

 

                                                      for (int j = -7; j < 7; j++)

                                                      {

                                                               // 为了 良好的用户体验 撕一个圆

                                                               if (Math.sqrt(i * i + j * j) < 7)

                                                               {

                                                                        // 一次修改一个像素

                                                                        try

                                                                        {

                                                                                 alterBitmap.setPixel((int) event.getX() + i, (int) event.getY() + j,

                                                                                                   Color.TRANSPARENT);// TRANSPARENT透明

 

                                                                        } catch (Exception e)

                                                                        {

                                                                        }

                                                               }

 

                                                      }

 

                                             }

 

                                             // 一定要记得更新iv

                                             iv.setImageBitmap(alterBitmap);

                                             break;

 

                                    }

 

                                    return true;

                           }

//音乐播放服务

public class MusicService extends Service

{

 

         private MediaPlayer player;

 

         // [2]把我们定义的中间人对象 返回

         @Override

         public IBinder onBind(Intent intent)

         {

                  return new MyBinder();

         }

 

         // 服务第一次开启的是调用

         @Override

         public void onCreate()

         {

 

                  // [1]初始化mediaplayer

                  player = new MediaPlayer();

 

                  super.onCreate();

         }

 

         // 当服务销毁的时候调用

         @Override

         public void onDestroy()

         {

                  super.onDestroy();

         }

 

         // 把seekBar指定的进度设置给播放器，从而从这个指定的位置播放

         public void seekToPosition(int position)

         {

                  player.seekTo(position);

         }

 

         // 专门用来播放音乐的

         public void playMusic()

         {

                  System.out.println("音乐播放了");

 

                  // [2]设置要播放的资源 path 可以是本地也可是网络路径

                  try

                  {

                           player.reset();

 

                           player.setDataSource("/mnt/sdcard/xpg.mp3");

 

                           // [3]准备播放

                           player.prepare();

 

                           // [4]开始播放

                           player.start();

 

                           // [5]更新进度条

                           updateSeekBar();

 

                  } catch (Exception e)

                  {

                           e.printStackTrace();

                  }

 

         }

 

         // 更新进度条的方法

         private void updateSeekBar()

         {

                  // [1]获取当前歌曲总时长

                  final int duration = player.getDuration();

 

                  // [2]一秒钟获取一次当前进度

                  final Timer timer = new Timer();

 

                  final TimerTask task = new TimerTask()

                  {

 

                           @Override

                           public void run()

                           {

                                    // [3]获取当前歌曲的进度

                                    int currentPosition = player.getCurrentPosition();

 

                                    // [4]创建message对象

                                    Message msg = Message.obtain();

 

                                    // [5]使用msg携带多个数据。msg.what只能携带一个数据

                                    Bundle bundle = new Bundle();

                                    bundle.putInt("duration", duration);

                                    bundle.putInt("currentPosition", currentPosition);

 

                                    msg.setData(bundle);

 

                                    // 发送消息 MainActivity的handlemessage方法会执行

                                    // 在MainActivity中handler为静态的，所以可以直接调用

                                    MainActivity.handler.sendMessage(msg);

 

                           }

                  };

                  // 300毫秒后 每隔1秒钟获取一次当前歌曲的进度

                  timer.schedule(task, 300, 1000);

 

                  // [3]当歌曲播放完成的时候 把timer和task取消

                  player.setOnCompletionListener(new OnCompletionListener()

                  {

 

                           // 当歌曲播放完成的回调

                           @Override

                           public void onCompletion(MediaPlayer mp)

                           {

                                    System.out.println("歌曲播放完成了 ");

 

                                    timer.cancel();

                                    task.cancel();

 

                           }

                  });

 

         }

 

         // 音乐暂停了

         public void pauseMusic()

         {

                  System.out.println("音乐暂停了");

                  // 暂停

                  player.pause();

 

         }

 

         // 音乐继续播放的方法

         public void rePlayMusic()

         {

                  System.out.println("音乐继续播放了");

 

                  player.start();

 

         }

 

         // [1]定义一个中间人对象(IBinder)

         private class MyBinder extends Binder implements Iservice

         {

 

                  // 调用播放音乐的方法

                  @Override

                  public void callPlayMusic()

                  {

 

                           playMusic();

                  }

 

                  // 调用暂停音乐的方法

                  @Override

                  public void callPauseMusic()

                  {

 

                           pauseMusic();

                  }

 

                  // 调用继续播放的方法

                  @Override

                  public void callrePlayMusic()

                  {

 

                           rePlayMusic();

                  }

 

                  // 调用设置播放指定位置的方法

                  @Override

                  public void callSeekToPosition(int position)

                  {

 

                           seekToPosition(position);

                  }

 

         }

 

}
