第一集  DragLayout的实现

 

先来实现一个自定义控件；后面在此基础上学习QQ 5.0 新特性。

 

/**

 * 第一步：创建包含主界面和左侧界面的DragLayout，首先继承自ViewGroup,后续步骤会修改为继承自FrameLayout；

 * 实现它的三个构造方法和onLayout方法

 */

public class DragLayout extends FrameLayout

{

 

        /**

         * 第二步：创建一个TestActivity，其界面中如下：

         *

         * <com.itheima74.qqslidemenu.test.DragLayout ... >

         *

         * <TextView ..redView.. />

         *

         * <TextView ..blueView../>

         *

         * </com.itheima74.qqslidemenu.test.DragLayout>

         */

 

        private View redView;// 红孩子

        private View blueView;// 蓝精灵

 

        private ViewDragHelper viewDragHelper;

 

        public DragLayout(Context context, AttributeSet attrs, int defStyle)

        {

                super(context, attrs, defStyle);

                init();

        }

 

        public DragLayout(Context context, AttributeSet attrs)

        {

                super(context, attrs);

                init();

        }

 

        public DragLayout(Context context)

        {

                super(context);

                init();

        }

 

        /**

         * 第六步：现在要实现对两个子控件的拖拽移动。一般情况下对某个控件的拖拽移动有以下三种方法：

         *

         * 1.通过改变view在父View的layout位置来移动,但是只能移动指定的View: view.layout(l,t,r,b);

         * view.offsetLeftAndRight(offset);//同时改变left和right

         * view.offsetTopAndBottom(offset);//同时改变top和bottom

         *

         * 2.通过改变scrollX和scrollY来移动,但是可以移动所有的子View； scrollTo(x,y);

         * scrollBy(xOffset,yOffset);

         *

         * 3.通过改变Canvas绘制的位置来移动View的内容: canvas.drawBitmap(bitmap, left, top, paint)

         *

         * 现在来学习使用一个高级的类ViewDragHelper来实现对控件的拖拽移动：

         *

         * 1.ViewDragHelper在android 4.4以上的v4包中才有

         * 2.它主要用于处理ViewGroup中对子View的拖拽处理

         * 3.它主要封装了对View的触摸位置，触摸速度，移动距离等的检测和Scroller,通过接口回调的方式供我们调用;

         * 4.android.support.v4.widget.ViewDragHelper

         *

         */

        private void init()

        {

                // this:谁的字控件，这里就是本类自定义控件

                // callback：回调

                viewDragHelper = ViewDragHelper.create(this, callback);

        }

 

        /**

         * 第三步：找到两个子控件。 在使用本自定义控件的xml(TestActivity的界面布局文件)中，当本控件的结束标签被读取完成会执行该方法；

         * 此时会知道自己有几个子View。 一般用于初始化子View的引用。 此方法只有继承自ViewGroup的自定义控件才有。

         */

        @Override

        protected void onFinishInflate()

        {

                super.onFinishInflate();

 

                redView = getChildAt(0);

                blueView = getChildAt(1);

 

        }

 

        /**

         * 第四步：测量两个子控件

         */

        @Override

        protected void onMeasure(int widthMeasureSpec, int heightMeasureSpec)

        {

                super.onMeasure(widthMeasureSpec, heightMeasureSpec);

 

// 第一种方式：如果界面布局文件中子控件宽高不确定时，一定要实

现onMeasure方法，并采用这种方式

                int measureSpec = MeasureSpec.makeMeasureSpec(

                                redView.getLayoutParams().width, MeasureSpec.EXACTLY);

 

                redView.measure(measureSpec, measureSpec);

                blueView.measure(measureSpec, measureSpec);

 

// 第二种方式：如果界面布局文件中子控件宽高已经确定时可以使用//这种简化方式

                // measureChild(redView, widthMeasureSpec, heightMeasureSpec);

                // measureChild(blueView, widthMeasureSpec, heightMeasureSpec);

 

                // 第三种方式：

                // 在自定义ViewGroup的时候，如果界面布局文件中子控件宽高已经确定时，那么可以直接继承自系统已有的

                // 布局(比如FrameLayout)，目的是为了让已有的布局帮我们实行onMeasure，我们自己无需再重写onMeasure方法

                // 本类继承自FrameLayout，则此onMeasure方法都可以完全去掉了

 

        }

 

        /**

         * 第五步：摆放两个子控件。此时运行程序可以查看两个子控件在TestActivity中的效果

         */

        @Override

        protected void onLayout(boolean changed, int l, int t, int r, int b)

        {

                int left = getPaddingLeft();

                int top = getPaddingTop();

 

                redView.layout(left, top, left + redView.getMeasuredWidth(), top

                                + redView.getMeasuredHeight());

 

                // blueView在redView下方

                blueView.layout(left, redView.getBottom(),

                                left + blueView.getMeasuredWidth(), redView.getBottom()

                                                + blueView.getMeasuredHeight());

 

                // getHeight和getMeasuredHeight的区别:

                // getMeasuredHeight:只要view执行完onMeasure方法就能够获取到值；

                // getHeight：只有view执行完layout才能获取到值;

        }

 

        /**

         * 第八步：在onTouchEvent和onInterceptTouchEvent方法中设置回调对象的处理

         */

        @Override

        public boolean onTouchEvent(MotionEvent event)

        {

                // 将触摸事件交给ViewDragHelper来解析处理

                viewDragHelper.processTouchEvent(event);

 

                return true;

        }

 

        @Override

        public boolean onInterceptTouchEvent(MotionEvent ev)

        {

                // 让ViewDragHelper帮我们判断是否应该拦截

                boolean result = viewDragHelper.shouldInterceptTouchEvent(ev);

 

                return result;

        }

 

        /**

         * 第七步：创建ViewDragHelper的回调对象

         */

        private ViewDragHelper.Callback callback = new ViewDragHelper.Callback()

        {

                /**

                 * 第九步：实现tryCaptureView方法；Capture：捕捉。 用于判断是否捕获当前child的触摸事件

                 * child:当前触摸的子View

                 * return: true:捕获并解析 ； false：不处理

                 */

                @Override

                public boolean tryCaptureView(View child, int pointerId)

                {

                        return child == blueView || child == redView;

                }

 

                /**

                 * 第十步：当上面这个方法指定的子view被开始捕获和解析时调用。此方法在这里没啥用 capturedChild:当前被捕获的子view

                 */

                @Override

                public void onViewCaptured(View capturedChild, int activePointerId)

                {

                        super.onViewCaptured(capturedChild, activePointerId);

 

                        // Log.e("tag", "capturedChild");

                }

 

                /**

                 * 以下四步都在设置类似的事情：

                 *

                 * 第十一步：获取view水平方向的拖拽范围。

                 *

                 * 当子控件位于屏幕中间的时候，手指抬起，子控件回到左边沿或者右边沿的时候，计算速度时间关系就会用到这个值，看最后的效果就能明白了。

                 * 最好不要返回0

                 */

                @Override

                public int getViewHorizontalDragRange(View child)

                {

                        return getMeasuredWidth() - child.getMeasuredWidth();

                }

 

                /**

                 * 第十二步：获取view垂直方向的拖拽范围，同上一个方法。最好不要返回0

                 */

                public int getViewVerticalDragRange(View child)

                {

                        return getMeasuredHeight() - child.getMeasuredHeight();

                };

 

                /**

                 * 第十三步：控制child在水平方向的移动 的left值:表示ViewDragHelper认为你想让当前child的left改变的值：

                 * left=chile.getLeft()+dx dx:本次child水平方向移动的距离

                 * return:表示你真正想让child的left变成的值

                 *

                 * 其实就是设置子控件的移动范围

                 */

                @Override

                public int clampViewPositionHorizontal(View child, int left, int dx)

                {

                        if (left < 0)

                        {

                                // 限制左边界

                                left = 0;

 

                        } else if (left > (getMeasuredWidth() - child.getMeasuredWidth()))

                        {

                                // 限制右边界

                                left = getMeasuredWidth() - child.getMeasuredWidth();

                        }

 

                        return left;

                }

 

                /**

                 * 第十四步：控制child在垂直方向的移动 top值: 表示ViewDragHelper认为你想让当前child的top改变的值,

                 * top=chile.getTop()+dy dy:本次child垂直方向移动的距离

                 * return: 表示你真正想让child的top变成的值

                 *

                 * 其实就是设置子控件的移动范围

                 */

                public int clampViewPositionVertical(View child, int top, int dy)

                {

                        if (top < 0)

                        {

                                top = 0;

 

                        } else if (top > getMeasuredHeight() - child.getMeasuredHeight())

                        {

                                top = getMeasuredHeight() - child.getMeasuredHeight();

                        }

 

                        return top;

                };

 

                /**

                 * 第十五步： 当child的位置改变的时候执行,一般用来做其他子View的伴随移动 ，使几个View一起移动。

                 * changedView：位置改变的child

                 * left：child当前最新的left

                 * top: child当前最新的top

                 * dx: 本次水平移动的距离

                 * dy: 本次垂直移动的距离

                 */

                @Override

                public void onViewPositionChanged(View changedView, int left, int top,

                                int dx, int dy)

                {

                        super.onViewPositionChanged(changedView, left, top, dx, dy);

 

                        if (changedView == blueView)

                        {

                                // blueView移动的时候需要让redView跟随移动

                                redView.layout(redView.getLeft() + dx, redView.getTop() + dy,

                                                redView.getRight() + dx, redView.getBottom() + dy);

 

                        } else if (changedView == redView)

                        {

                                // redView移动的时候需要让blueView跟随移动

                                blueView.layout(blueView.getLeft() + dx,

                                                blueView.getTop() + dy, blueView.getRight() + dx,

                                                blueView.getBottom() + dy);

                        }

 

                       

                        /**

                         *  第十七步：计算view左右移动的百分比：移动过程中的左边距与移动到最右边时的左边距的百分比

                         */

                        float fraction = changedView.getLeft() * 1f

                                        / (getMeasuredWidth() - changedView.getMeasuredWidth());

                        executeAnim(fraction);

                }

 

                /**

                 *  第十六步：手指抬起的时候执行该方法

                       releasedChild：手指在releasedChild控件上抬起

                       xvel: x方向的移动的速度 正：向右移动， 负：向左移动

                 *  yvel: y方向移动的速度

                 */

                @Override

                public void onViewReleased(View releasedChild, float xvel, float yvel)

                {

                        super.onViewReleased(releasedChild, xvel, yvel);

                       

                        int centerLeft = getMeasuredWidth() / 2

                                        - releasedChild.getMeasuredWidth() / 2;

                        //屏幕宽的一半减去控件宽的一半，把控件移到屏幕中线就能看到其意义。以此作为判断控件在屏幕左半部分还是右半部分

                       

                        if (releasedChild.getLeft() < centerLeft)

                        {

                                // 在左半边，应该向左缓慢移动，直到屏幕左边线为止

                                // 两个参数分别是最终移动到的位置

                                // smooth:平滑

                                viewDragHelper.smoothSlideViewTo(releasedChild, 0,

                                                releasedChild.getTop());

 

                                //对DragLayout进行界面刷新。然后继续调用computeScroll方法进行循环，直到滚到终点位置

                                ViewCompat.postInvalidateOnAnimation(DragLayout.this);

 

                        } else

                        {

                                // 在右半边，应该向右缓慢移动，直到屏幕右边线为止

                                // 两个参数分别是最终移动到的位置

                                viewDragHelper.smoothSlideViewTo(releasedChild,

                                                getMeasuredWidth() - releasedChild.getMeasuredWidth(),

                                                releasedChild.getTop());

 

                                ViewCompat.postInvalidateOnAnimation(DragLayout.this);

 

                        }

                }

        };

 

        public void computeScroll()

        {

                //true表示动画还没结束，就继续滚动并刷新界面

                if (viewDragHelper.continueSettling(true))

                {

                        ViewCompat.postInvalidateOnAnimation(DragLayout.this);

                }

        }

       

       

        /**

         *  第十八步：执行伴随动画

         * 

         *  ViewHelper类需要导入nineoldandroids.jar

         */

        private void executeAnim(float fraction)

        {

                // fraction范围: 0 - 1

               

                // 缩放：在原有的基础之上放大0.5倍

                // ViewHelper.setScaleX(redView, 1+0.5f*fraction);

                // ViewHelper.setScaleY(redView, 1+0.5f*fraction);

               

                // 旋转

                // ViewHelper.setRotation(redView,360*fraction);//围绕中心旋转

                //ViewHelper.setRotationX(redView, 360 * fraction);// 围绕x轴上下翻转

                 ViewHelper.setRotationY(redView,360*fraction);//围绕y左右翻转

 

                // 平移

                // ViewHelper.setTranslationX(redView,80*fraction);

               

                // 透明

                // ViewHelper.setAlpha(redView, 1-fraction);

 

                /**

                 *  第十九步：设置过度颜色的渐变。ColorUtil直接复制过来用。

                 *  参数表示从红色变为绿色

                 */

                redView.setBackgroundColor((Integer) ColorUtil.evaluateColor(fraction,

                                Color.RED, Color.GREEN));

       

        }

 

}

/**

 * 第一步：新建自定义控件SlideMenu，继承自FrameLayout；实现其三个构造方法

 */

public class SlideMenu extends FrameLayout

{

 

        /**

         * 第二步：在MainActivity的界面布局文件activity_main如下：

         *

         * <com.itheima74.qqslidemenu.SlideMenu ...

         * android:background="@drawable/bg" //设置一张背景图 ... >//根节点

         *

         * <!-- 侧滑菜单的布局 --> <include layout="@layout/layout_menu"/> 如下：

         *

              

 

         * <!-- 主界面的布局 --> <include layout="@layout/layout_main"/> 如下：

 

         *

         * </com.itheima74.qqslidemenu.SlideMenu>

         */

 

        private View mainView;// 主界面的view

        private View leftMenuView;// 侧滑菜单的view

 

        private ViewDragHelper viewDragHelper;

 

        private int width;

 

        private float dragRange;// 拖拽范围

 

        private FloatEvaluator floatEvaluator;// float的计算器

        private IntEvaluator intEvaluator;// int的计算器

 

        public SlideMenu(Context context, AttributeSet attrs, int defStyle)

        {

                super(context, attrs, defStyle);

                init();

        }

 

        public SlideMenu(Context context, AttributeSet attrs)

        {

                super(context, attrs);

                init();

        }

 

        public SlideMenu(Context context)

        {

                super(context);

                init();

        }

 

        /**

         * 第十六步： 定义侧滑菜单打开关闭两个状态和默认值

         */

        enum DragState

        {

                Open, Close;

        }

 

        private DragState currentState = DragState.Close;// 当前SlideMenu的状态默认是关闭的

 

        /**

         * 第四步：创建init方法；创建viewDragHelper对象

         */

        private void init()

        {

                viewDragHelper = ViewDragHelper.create(this, callback);

 

                floatEvaluator = new FloatEvaluator();

                intEvaluator = new IntEvaluator();

        }

 

        /**

         * 获取当前的侧滑菜单的状态；用于MyLinearLayout中进行判断

         */

        public DragState getCurrentState()

        {

                return currentState;

        }

 

        /**

         * 第三步：找到activity_main中的侧滑菜单和主界面两个子控件

         */

        @Override

        protected void onFinishInflate()

        {

                super.onFinishInflate();

 

                // 简单的异常处理：本自定义控件只能有两个子控件

                if (getChildCount() != 2)

                {

                        throw new IllegalArgumentException(

                                        "SlideMenu only have 2 children!");

                }

 

                leftMenuView = getChildAt(0);

                mainView = getChildAt(1);

        }

 

        /**

         * 第八步：该方法在onMeasure执行完之后执行；可以在该方法中初始化自己和子View的宽高

         */

        @Override

        protected void onSizeChanged(int w, int h, int oldw, int oldh)

        {

                super.onSizeChanged(w, h, oldw, oldh);

 

                // 获取本自定义控件SlideMenu的宽度

                width = getMeasuredWidth();

 

                dragRange = width * 0.6f;

                // 这里设置的是主界面往右滑动时，左边遗留给侧滑界面显示的宽度

        }

 

        /**

         * 第五步：在onTouchEvent和onInterceptTouchEvent方法中设置相关操作

         */

        @Override

        public boolean onTouchEvent(MotionEvent event)

        {

                viewDragHelper.processTouchEvent(event);

 

                return true;

        }

 

        @Override

        public boolean onInterceptTouchEvent(MotionEvent ev)

        {

                return viewDragHelper.shouldInterceptTouchEvent(ev);

        }

 

        /**

         * 第六步：创建viewDragHelper的回调对象

         */

        private ViewDragHelper.Callback callback = new ViewDragHelper.Callback()

        {

                /**

                 * 第七步：用于判断是否捕获当前child的触摸事件 child: 当前触摸的子View return: true:就捕获并解析

                 * false：不处理

                 */

                @Override

                public boolean tryCaptureView(View child, int pointerId)

                {

                        return child == leftMenuView || child == mainView;

                }

 

                /**

                 * 以下四步都在设置类似的事情：

                 *

                 * 第九步：获取view水平方向的拖拽范围。

                 *

                 * 当子控件位于屏幕中间的时候，手指抬起，子控件回到左边沿或者右边沿的时候，计算速度时间关系就会用到这个值，看最后的效果就能明白了。

                 * 最好不要返回0

                 */

                public int getViewHorizontalDragRange(View child)

                {

                        return (int) dragRange;

                }

 

                /**

                 * 第十步：控制child在水平方向移动范围 left: 表示ViewDragHelper认为你想让当前child的left改变的值,

                 * left=chile.getLeft()+dx dx:本次child水平方向移动的距离

                 * return:表示你真正想让child的left变成的值

                 *

                 */

                public int clampViewPositionHorizontal(View child, int left, int dx)

                {

                        if (child == mainView)

                        {

                                if (left < 0)

 

                                        left = 0;// 限制mainView的左边

 

                                if (left > dragRange)

 

                                        left = (int) dragRange;// 限制mainView的右边

                        }

 

                        return left;

                }

 

                /**

                 * 第十一步：当child的位置改变的时候执行,一般用来做其他子View的伴随移动 ，使几个View一起移动。

                 * changedView：位置改变的child left：child当前最新的left top:child当前最新的top

                 * dx:本次水平移动的距离 dy: 本次垂直移动的距离

                 *

                 */

                public void onViewPositionChanged(View changedView, int left, int top,

                                int dx, int dy)

                {

                        if (changedView == leftMenuView)

                        {

                                // 固定住leftMenuView

                                leftMenuView.layout(0, 0, leftMenuView.getMeasuredWidth(),

                                                leftMenuView.getMeasuredHeight());

 

                                // 让mainView移动起来

                                // 限制mainView的边界

                                int newLeft = mainView.getLeft() + dx;

                                if (newLeft < 0)

                                {

                                        newLeft = 0;

                                }

                                if (newLeft > dragRange)

                                {

 

                                        newLeft = (int) dragRange;

                                }

 

                                mainView.layout(newLeft, mainView.getTop() + dy, newLeft

                                                + mainView.getMeasuredWidth(), mainView.getBottom()

                                                + dy);

                        }

 

                        /**

                         * 第十四步：设置主界面和侧滑界面拖动过程中的一系列动画

                         */

 

                        // 第十四步：1.计算滑动的百分比

                        float fraction = mainView.getLeft() / dragRange;

 

                        // 第十四步： 2.执行伴随动画

                        executeAnim(fraction);

 

                        /**

                         * 第十七步：更改状态，回调listener的方法

                         */

                        // 第十八步见MainActivity

                        if (fraction == 0 && currentState != DragState.Close)

                        {// 之所以要加上后面一个判断条件，是为了防止用户刚开启侧边栏一点点，然后松手，侧边栏又关闭这种情况

 

                                // 更改状态为关闭，并回调关闭的方法

                                currentState = DragState.Close;

 

                                if (listener != null)

                                {

                                        listener.onClose();

                                }

 

                        } else if (fraction == 1f && currentState != DragState.Open)

                        {

                                // 更改状态为打开，并回调打开的方法

                                currentState = DragState.Open;

 

                                if (listener != null)

                                {

                                        listener.onOpen();

                                }

                        }

 

                        // 将drag的fraction暴漏给外界MainActivity

                        if (listener != null)

                        {

                                listener.onDraging(fraction);

                        }

                }

 

                /**

                 * 第十二步：手指抬起的时候执行该方法 releasedChild：手指在releasedChild控件上抬起 xvel: x方向的移动的速度

                 * 正：向右移动， 负：向左移动 yvel: y方向移动的速度

                 */

                public void onViewReleased(View releasedChild, float xvel, float yvel)

                {

                        if (mainView.getLeft() < dragRange / 2)

                        {

                                // 在左半边

                                close();

 

                        } else

                        {

                                // 在右半边

                                open();

                        }

 

                        /**

                         * 第二十一步：处理用户的稍微滑动：用户在短距离内快速滑动

                         */

                        if (xvel > 200 && currentState != DragState.Open)

                        {

                                open();

 

                        } else if (xvel < -200 && currentState != DragState.Close)

                        {

                                close();

                        }

                }

        };

 

        /**

         * 属于第十二步：关闭菜单

         */

        public void close()

        {

                viewDragHelper.smoothSlideViewTo(mainView, 0, mainView.getTop());

 

                ViewCompat.postInvalidateOnAnimation(SlideMenu.this);

        }

 

        /**

         * 属于第十二步： 打开菜单

         */

        public void open()

        {

                viewDragHelper.smoothSlideViewTo(mainView, (int) dragRange,

                                mainView.getTop());

 

                ViewCompat.postInvalidateOnAnimation(SlideMenu.this);

        }

 

        // 属于第十二步： 别忘了实现这个方法

        // 第十三步：见MainActivity

        public void computeScroll()

        {

                if (viewDragHelper.continueSettling(true))

                {

                        ViewCompat.postInvalidateOnAnimation(SlideMenu.this);

                }

        };

 

        /**

         * 第十四步： 2.执行伴随动画

         */

        private void executeAnim(float fraction)

        {

                // fraction:0-1

 

                // 从左往右拖动时缩小mainView：从1缩小至0.8

                // 在init方法中创建floatEvaluator对象，其来自nineoldandroids包

                ViewHelper.setScaleX(mainView,

                                floatEvaluator.evaluate(fraction, 1f, 0.8f));

                ViewHelper.setScaleY(mainView,

                                floatEvaluator.evaluate(fraction, 1f, 0.8f));

 

                // 从左往右拖动时设置leftMenuView动画

                // 在init方法中创建intEvaluator对象，其来自nineoldandroids包

                // 平移动画

                ViewHelper.setTranslationX(

                                leftMenuView,

                                intEvaluator.evaluate(fraction,

                                                -leftMenuView.getMeasuredWidth() / 2, 0));

                // 缩放动画

                ViewHelper.setScaleX(leftMenuView,

                                floatEvaluator.evaluate(fraction, 0.5f, 1f));

                ViewHelper.setScaleY(leftMenuView,

                                floatEvaluator.evaluate(fraction, 0.5f, 1f));

 

                // 透明动画

                ViewHelper.setAlpha(leftMenuView,

                                floatEvaluator.evaluate(fraction, 0.3f, 1f));

 

                // 给SlideMenu的背景图片添加黑色的遮罩效果。getBackground()方法是获取SlideMenu的背景图片，在这张背景图片上进行颜色设置。

                // 此图片即activity_main.xml中根布局的背景图片

                // ColorUtil直接复制使用

                getBackground().setColorFilter(

                                (Integer) ColorUtil.evaluateColor(fraction, Color.BLACK,

                                                Color.TRANSPARENT), Mode.SRC_OVER);

        }

 

        /**

         * 第十五步： 提供接口，把打开，关闭和拖动百分比传递给外界MainActivity使用

         */

        public interface OnDragStateChangeListener

        {

                /**

                 * 打开的回调

                 */

                void onOpen();

 

                /**

                 * 关闭的回调

                 */

                void onClose();

 

                /**

                 * 正在拖拽中的回调

                 */

                void onDraging(float fraction);

        }

 

        private OnDragStateChangeListener listener;

 

        public void setOnDragStateChangeListener(OnDragStateChangeListener listener)

        {

                this.listener = listener;

        }

}

/**

 * 第十九步：创建MyLinearLayout类，继承自LinearLayout；将主界面的layout_main.xml中的根布局替换为此线性布局；

 * 当slideMenu打开的时候，拦截并消费掉主界面的任何触摸事件， 这个时候主界面设置为不能点击；当在主界面上点击无效抬起的时候，关闭侧滑菜单

 */

// 第二十步见MainActivity

public class MyLinearLayout extends LinearLayout

{

        public MyLinearLayout(Context context, AttributeSet attrs, int defStyle)

        {

                super(context, attrs, defStyle);

        }

 

        public MyLinearLayout(Context context, AttributeSet attrs)

        {

                super(context, attrs);

        }

 

        public MyLinearLayout(Context context)

        {

                super(context);

        }

 

        // 把slideMenu传递进来

        private SlideMenu slideMenu;

 

        public void setSlideMenu(SlideMenu slideMenu)

        {

                this.slideMenu = slideMenu;

        }

 

        @Override

        public boolean onInterceptTouchEvent(MotionEvent ev)

        {

                if (slideMenu != null && slideMenu.getCurrentState() == DragState.Open)

                {

                        // 如果slideMenu打开则应该拦截并消费掉事件

                        return true;

                }

 

                return super.onInterceptTouchEvent(ev);

        }

 

        @Override

        public boolean onTouchEvent(MotionEvent event)

        {

                if (slideMenu != null && slideMenu.getCurrentState() == DragState.Open)

                {

                        if (event.getAction() == MotionEvent.ACTION_UP)

                        {

                                // 抬起则应该关闭slideMenu

                                slideMenu.close();

                        }

 

                        // 如果slideMenu打开则应该拦截并消费掉事件

                        return true;

                }

 

                return super.onTouchEvent(event);

        }

}
