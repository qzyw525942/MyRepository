//整个类之前和这个类的各个方法前要加上注释

 

/**           

* 这是针对数组进行操作的工具类     //类的功能说明 

* @author 刘意                     //代码和注释的作者和版本号

* @version V.10

*/

public class ArrayTool {

      

       /**

       * 这是私有构造方法                //方法功能说明

       */

       private ArrayTool(){}

 

       /**

       * 这是遍历数组的方法

* @param arr 这是要被遍历的数组      //这里是方法要传入的参数说明

       */

       public static void printArray(int[] arr) {

              for(int x=0; x<arr.length; x++) {

                     if(x == arr.length-1) {

                            System.out.println(arr[x]);

                     }else {

                            System.out.print(arr[x]+", ");

                     }

              }

       }

      

       /**

       * 这是获取数组中最大值的方法

       * @param  arr 这是要获取最大值的数组

       * @return 返回数组中的最大值     //方法要返回的数据说明

       */

       public static int getMax(int[] arr) {

              int max = arr[0];

             

              for(int x=1; x<arr.length; x++) {

                     if(arr[x] > max) {

                            max = arr[x];

                     }

              }

             

              return max;

       }

      

       /**

       * 获取指定元素在数组中第一次出现的索引，如果元素不存在，就返回-1

       * @param arr 被查找的数组

       * @param value 要查找的元素

       * @return 返回元素在数组中的索引，如果不存在，返回-1

       */

       public static int getIndex(int[] arr,int value) {

              int index = -1;

             

              for(int x=0; x<arr.length; x++) {

                     if(arr[x] == value) {

                            index = x;

                            break;

                     }

              }

             

              return index;

       }

}

案例2：

      

猜数字小游戏(数据在1-100之间)

      

       分析：

              A:程序产生一个随机数。(被猜的)

              B:键盘录入数据。(你猜的)

              C:把你猜的和被猜的进行比较

                     a:大了

                     b:小了

                     c:猜中了

              D:给出多次猜的机会，猜中就结束。

                     while()循环，猜中就break

import java.util.Scanner;

 

class GuessNumber {

       public static void main(String[] args) {

              //程序产生一个随机数。(被猜的)

              int number = (int)(Math.random()*100)+1;//这部分代码不能在while语句中，产生一个随机数再来猜它，如果在while语句中则每一次都会产生新的随机数，难以猜中了

             

              //给出多次猜的机会，猜中就结束。

              while(true) {

                     //键盘录入数据。(你猜的)

                     Scanner sc = new Scanner(System.in);//这部分代码要在while循环语句中，每次循环就输入一个数字，再来和随机数比较

                     System.out.println("请输入你要猜的数据(1-100):");

                     int guessNumber = sc.nextInt();

                    

                     //把你猜的和被猜的进行比较

                     if(guessNumber > number) {

                            System.out.println("你猜的数据"+guessNumber+"大了");

                     }else if(guessNumber < number) {

                            System.out.println("你猜的数据"+guessNumber+"小了");

                     }else {

                            System.out.println("恭喜你，猜中了");

                            break;//这个break是属于while循环的，if语句不能有break

                     }

              }

       }

}

 

第三集     代码块

 

在Java中，使用{}括起来的代码被称为代码块。

 

              局部代码块:局部位置,用于限定变量的生命周期。变量只在此代码块内起作用

              构造代码块:与类中的成员变量和成员方法同级,用{}括起来的代码。每次调用构造                     方法执行前，都会先执行构造代码块。即每次创建对象前都会执行构造代码块

                     作用：可以把多个构造方法中的共同代码放到一起，对对象进行初始化。

              静态代码块:与类中的成员变量和成员方法同级,用{}括起来的代码,只不过它用                            static修饰了。

                     作用：一般是对类进行初始化。

              同步代码块：在多线程中讲解

      

class Code {

       static {  //静态代码块，随类的加载而执行且只执行一次，对类进行初始化

              int a = 1000;

              System.out.println(a);

       }

 

       //构造代码块，每次创建对象前都会执行，对对象进行初始化

       {

              int x = 100;

              System.out.println(x);

       }

 

  public Code(){  //构造方法

}

}

 

class CodeDemo {

       public static void main(String[] args) {

              //局部代码块，注意局部代码块在方法中，构造代码块与成员同级

              {

                     int x = 10;

                     System.out.println(x);

              }

             

              //System.out.println(x);//找不到符号，因为x只在局部代码块之内起作用

             

              Code c = new Code(); //先执行构造代码块，再执行构造方法

              System.out.println("---------------");

              Code c2 = new Code();//先执行构造代码块，再执行构造方法

              System.out.println("---------------");

              Code c3 = new Code(1);//先执行构造代码块，再执行构造方法

       }

}

第四集  继承

 

把多个类中相同的内容提取出来定义到一个类中。多个类再从这个类继承这些内容。

 

子类可以继承父类中的成员变量，成员方法，也可以扩展子类自己的成员变量和成员方法

 

继承的好处：

              A:减少了代码的重复性：重复的代码都在父类中了

              B:提高了代码的维护性：如果要对子类做修改，只需要修改父类即可

              C:让类与类之间产生关系，是多态的前提

      

       类与类产生了关系，其实也是继承的一个弊端：

              类的耦合性增强了。

             

              而开发的原则是：低耦合，高内聚。

              耦合：类与类的关系

              内聚：就是自己完成某件事情的能力

 

 

继承的特点：

 

A:Java只支持单继承，不支持多继承。

 

class Father {}

class Mother {}

class Son exnteds Father {} //正确的

class Son extends Father,Mother {} // 错误的

 

B:Java支持多层继承(继承体系)

 

class GrandFather {

       public void show() {

              System.out.println("我是爷爷");

       }

}

 

class Father extends GrandFather {

       public void method(){

              System.out.println("我是父亲");

       }

}

 

class Son extends Father {}

 

class ExtendsDemo2 {

       public static void main(String[] args) {

              Son s = new Son();

              s.method(); //使用父亲的

              s.show(); //使用爷爷的

       }

}

 

继承的注意事项：

       A:子类只能继承父类所有非私有的成员(成员方法和成员变量)

       B:子类不能继承父类的构造方法，但是可以通过super关键字去访问父类的构造方法。

 

class Father {

       private int num = 10;

       public int num2 = 20;

      

       private void method() {//私有方法，子类不能继承

              System.out.println(num);

              System.out.println(num2);

       }

      

       public void show() {//公共方法，子类可以继承

              System.out.println(num);//虽然子类不能继承父类的私有成员变量，但可以通过父                                           //类的公有方法去访问这个私有变量

              System.out.println(num2);

       }

}

 

class Son extends Father {

       public void function() {

              //System.out.println(num); //子类不能继承父类的私有成员变量

              System.out.println(num2);

       }

}

继承中父类的成员变量和子类的成员变量的关系：

 

A:子类中的成员变量和父类中的成员变量名称不一样，则子类能够区分出二者，且都能   用

    B:子类中的成员变量和父类中的成员变量名称一样

就近原则：

a:在子类方法的局部范围找，有就使用

b:在子类的成员范围找，有就使用

c:在父类的成员范围找，有就使用

d:如果还找不到，就报错。

 

 

class Father {

       public int num = 10;//父类的成员范围

      

       public void method() {//不调用这个方法就不会调用里面的num

              int num = 50;

       }

}

 

class Son extends Father {

       public int num = 30;//子类的成员范围

       public int num2 = 20;

 

       public void show() {

              int num = 40;//子类方法的局部位置

              System.out.println(num);//这个num按照就近原则，是40

              System.out.println(num2);//20

              System.out.println(num3);// 找不到符号，出错

 

       }

案例1：方法重写

 

方法重写：子类中出现的和父类中方法一模一样的方法。

              方法名一样，参数列表一样，返回值也一样。

方法重载：

              本类中出现的方法名一样，参数列表不同的方法。与返回值无关。

 

方法重写的应用：

       当子类需要父类的方法，又需要扩展父类的方法时，可以重写父类中的方法。这样，即沿袭了父类的方法，又定义了子类特有的功能。

 

class Phone {

       public void call(String name) {

              System.out.println("给"+name+"打电话");

       }

}

 

class NewPhone extends Phone {

       public void call(String name) {//子类的方法和父类的方法名一样，参数列表一样，返回值一样，

              super.call(name);//保留父类中与这个方法名一样的方法的功能

              System.out.println("可以听天气预报了");//子类再扩展自己的功能，

       }

}

 

class ExtendsDemo9 {

       public static void main(String[] args) {

              np.call("林青霞");//既可以打印出父类中的“给林青霞打电话”，又能打印出子类中                                        //的“可以听天气预报了”

       }

案例2：方法重写的注意事项

 

方法重写的注意事项

       A:父类中的私有方法不能被继承，覆盖，也就不能被重写

       如果父类有一个私有方法，子类也有一个同名的私有方法，则二者没有关系，父类的是父类的，子类的是子类的

       B:子类重写父类方法时，访问权限不能更低

       如父类的show方法为public，则子类的show方法的权限要大于等于public权限，最好等于

       C:父类方法为静态方法，重写方法时，子类也必须是静态方法

                    

       总结起来就是：子类重写父类方法的时候，声明最好一模一样。

 

两个面试题： 

 

1：方法重写和方法重载的区别?方法重载能改变返回值类型吗?

 

方法重写（Override）：

       在子类中，出现和父类中一模一样的方法声明的现象。

      

方法重载（Overload）：

       同一个类中，出现的方法名相同，参数列表不同的现象。

 

 

方法重载能改变返回值类型，因为它和返回值类型无关。

 

2：this关键字和super关键字分别代表什么?以及他们各自的使用场景和作用。

 

this:代表当前类的对象的引用

super:代表父类的引用

 

场景：

       成员变量：

              this.成员变量

              super.成员变量

       构造方法：

              this(...)

              super(...)

       成员方法：

              this.成员方法

              super.成员方法
