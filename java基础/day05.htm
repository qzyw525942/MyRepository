import java.util.Scanner;

 

class FunctionTest {

       public static void main(String args[]) {

              //创建键盘录入对象

              Scanner sc = new Scanner(System.in);

              int result = getMax(a,b);

              System.out.println("较大值是："+result);

       }

      

       /*

              需求：两个数中的较大值

              两个明确：

                     返回值类型：int

                     参数列表：int a,int b                    

       */

       public static int getMax(int a,int b) {

              if(a > b) {

                     return a;

              }else {

                     return b;

              }

              int c = ((a > b)? a: b);

              return c;

       }

}

import java.util.Scanner;

 

class FunctionTest5 {

                     }

              }

       }

}

 

方法重载：

多个方法的功能相同，即方法名字相同，但是参数列表不同的情况，就叫做方法重载。注意，方法重载只与方法名字和参数列表有关，与返回值类型无关。其中参数列表不同又分为参数个数的不同和参数类型的不同。

 

class FunctionDemo4 {

       public static void main(String[] args) {

 

              //JVM会根据所给的参数去调用不同的方法

 

              System.out.println(sum(10,20));

              System.out.println(sum(10,20,30));

              System.out.println(sum(10,20,30,40));

             

              System.out.println(sum(10.5f,20f));//调用的是float的方法

       }

      

       //需求1:求两个数的和

       public static int sum(int a,int b) {

              return a + b;

       }

      

       //需求2:求三数的和

       public static int sum(int a,int b,int c) {

              return a + b + c;

       }

      

       //需求3:求四个数的和

       public static int sum(int a,int b,int c,int d) {

              return a + b + c + d;

       }

      

       public static float sum(float a,float b) {

              return a + b;

       }

}

第二集          数组
数组:存储同一种数据类型的多个元素的一个容器。

定义格式：

              A:数据类型[] 数组名;

              B:数据类型 数组名[];

             

       举例：

              int[] a;        

int a[];        

      

二者的效果是一样的

 

数组的初始化：数组不初始化直接打印出会出错，提醒如下错误：

 

int[] a;

System.out.println(a);//错误提醒:可能尚未初始化变量a

 

两种初始化方法：

1、动态初始化：只指定长度，由系统给出初始化值

 

       动态初始化的格式：

              数据类型[] 数组名 = new 数据类型[数组长度];

             

              举例：

              int[] arr = new int[3];//3表示数组中能存储的元素个数

 

 

左边：

                            int:说明数组中的元素的数据类型是int类型

                            []:说明这是一个数组

                            arr:是数组的名称

                           

右边：

                            new:为数组分配内存空间。

                            int:说明数组中的元素的数据类型是int类型

                            []:说明这是一个数组

                            3:数组长度，其实也就是数组中元素的个数

 

如果此时：System.out.println(arr); //则会打印出：[I@175078b 内存地址值

 

如何获取数组中的元素呢?通过:

数组名[索引]

                     索引其实就是每个元素的编号，从0开始，最大索引是数组的长度-1。

 

System.out.println(arr[0]);//默认值是0，因为数组是int型的

              System.out.println(arr[1]);

案例2、获取数组的最值（最大值或者最小值）

 

class ArrayTest2 {

       public static void main(String[] args) {

              //定义一个数组

              int[] arr = {34,98,10,25,67};

             

              //调用方法

              int max = getMax(arr);

              System.out.println("max:"+max);

                    

              //请获取数组中的最小值

              int min = getMin(arr);

              System.out.println("min:"+min);

       }

       public static int getMax(int[] arr) {

              //从数组中任意的找一个元素作为参照物

              int max = arr[0];

              //然后遍历其他的元素

              for(int x=1; x<arr.length; x++) {//第0个已经作为了参照物，所以从第一个开始

                     //依次获取和参照物进行比较，如果大就留下来，如果小，就离开。

                     if(arr[x] > max) {

                            max = arr[x];

                     }

              }

              //最后参照物里面保存的就是最大值。

              return max;

       }

      

       public static int getMin(int[] arr) {

              //从数组中任意的找一个元素作为参照物

              int min = arr[0];

              //然后遍历其他的元素

              for(int x=1; x<arr.length; x++) {

                     //依次获取和参照物进行比较，如果小就留下来，如果大，就离开。

                     if(arr[x] < min) {

                            min = arr[x];

                     }

              }

              //最后参照物里面保存的就是最小值。

              return min;

       }

}

 

案例3、数组元素逆序（两种方式）

 

分析：

              A:定义一个数组，并进行静态初始化。

              B:思路

                     把0索引和arr.length-1的数据交换

                     把1索引和arr.length-2的数据交换

                     ...

                     只要调到arr.length/2的时候即可，无论是奇数个还是偶数个，

                     因为奇数个的话中间那个数不动，偶数个的话中间的两个数调换。

 

class ArrayTest3 {

       public static void main(String[] args) {

              //定义一个数组，并进行静态初始化。

              int[] arr = {12,98,50,34,76};

 

//reverse(arr);

              reverse2(arr);

              printArray(arr);

}

 

public static void reverse(int[] arr) {//有人会想到应该返回的是逆序后的数组，但是没必要，因为逆序前的和逆序后的两个数组其实是同一个数组，不过是修改了顺序了而已)

for(int x=0; x<arr.length/2; x++) {

                     int temp = arr[x];

                     arr[x] = arr[arr.length-1-x];//arr[x]和 arr[arr.length-1-x]分别从数组                                                                的两边向中间缩进

                     arr[arr.length-1-x] = temp;

              }

       }

 

也可以是如下方式：

public static void reverse2(int[] arr) {

              for(int start=0,end=arr.length-1; start<=end; start++,end--) {//start和                                                                                     //end分别从两向中间缩进

                     int temp = arr[start];

                     arr[start] = arr[end];

                     arr[end] = temp;

              }

       }

案例5、数组元素查找(查找指定元素第一次在数组中出现的索引)

分析：

              A:定义一个数组，并静态初始化。

              B:写一个功能实现

                     遍历数组，依次获取数组中的每一个元素，和已知的数据进行比较

                     如果相等，就返回当前元素的索引值。

 

class ArrayTest5 {

       public static void main(String[] args) {

              //定义一个数组，并静态初始化

              int[] arr = {200,250,38,888,444};

             

              //需求：我要查找250在这个数组中第一次出现的索引

              int index = getIndex(arr,250);

              System.out.println("250在数组中第一次出现的索引是："+index);

             

              int index3 = getIndex2(arr,2500);

              System.out.println("2500在数组中第一次出现的索引是："+index3);

       }

      

       /*

                     两个明确：

                     返回值类型：int

                     参数列表：int[] arr,int value

       */

       public static int getIndex(int[] arr,int value) {

              //遍历数组，依次获取数组中的每一个元素，和已知的数据进行比较

              for(int x=0; x<arr.length; x++) {

                     if(arr[x] == value) {

                            //如果相等，就返回当前的索引值。

                            return x;

                     }

              }

              //如果找不到数据，我们一般返回一个负数即可，而且是返回-1（注意这是返回值            为int时是这样，其他的不一定）

              return -1;

       }

}
