//工具类

public class Student {

         private String name;

         private int age;

 

         public Student() {

                   super();

         }

 

         public Student(String name, int age) {

                   super();

                   this.name = name;

                   this.age = age;

         }

 

         public String getName() {

                   return name;

         }

 

         public void setName(String name) {

                   this.name = name;

         }

 

         public int getAge() {

                   return age;

         }

 

         public void setAge(int age) {

                   this.age = age;

         }

 

         @Override

         public boolean equals(Object obj) {//为什么要添加此方法见测试类的说明

                   if (this == obj)

                            return true;

                   if (obj == null)

                            return false;

                   if (getClass() != obj.getClass())

                            return false;

                   Student other = (Student) obj;

                   if (age != other.age)

                            return false;

                   if (name == null) {

                            if (other.name != null)

                                     return false;

                   } else if (!name.equals(other.name))

                            return false;

                   return true;

         }

 

}

//但这不是题目的意思，题目的意思是：自己定义一个集合类，在这个集合类内      //部可以使用LinkedList模拟。第一次见到这样的做法，如下：

 

 

/**

 * 自定义的栈集合

 */

public class MyStack

{

         private LinkedList link;

 

         public MyStack()

         {// 这个方法的意思是，

                   // 调用这个无参构造方法创建MyStack类的对象时，其实创建的是                                //LinkedList类的对象

                   link = new LinkedList();

         }

 

         public void add(Object obj)

         {// 这个方法的意思是，

                   // 调用这个add方法时，其实调用的是LinkedList类的addFirst方法，后                      //面的做法类似

                   link.addFirst(obj);

         }

 

         public Object get()

         {

                   // return link.getFirst();

                   // 因为是栈，所以这儿不能用getFirst，这个方法只能得到栈数据结构的                    //第一个元素,后面的元素不能得到

                   return link.removeFirst();

                   // 使用removeFirst就不一样，它能删除栈的第一个元素并把它返回，而                     //删除第一个元素后后面的元素就能弹上来，

                   // 下一次获取时就能得到后面的元素了，这就叫做弹栈

         }

 

         public boolean isEmpty()

         {

                   return link.isEmpty();

         }

}

     (1)泛型概述

                   是一种把明确集合类型的工作放到创建对象或者调用方法的时候才去明确的特殊              的类型。

     (2)格式：

                   <数据类型>

                   注意：该数据类型只能是引用类型，因为集合中只能存入引用类型。

     (3)好处：

                   A:把运行期间的问题提前到了编译期间

                   B:避免了强制类型转换

                   C:优化了程序设计，解决了黄色警告线问题，让程序更安全

 

运用泛型后的代码如下：

                   ArrayList<Integer> array = new ArrayList<Integer>();

 

                   // 以0结束,这个简单，只要键盘录入的数据是0，我就不继续录入数据了

                   while (true)

                   {// 判断条件先设为true

                            System.out.println("请输入数据：");

                            int number = sc.nextInt();

                            if (number != 0)

                            {

                                     array.add(number);

                            } else

                            {

                                     break;// 一旦遇到0就终止循环

                            }

                   }

 

                   // 把集合转成数组：用另一个方法：

                   // public <T> T[] toArray(T[] a)

                   Integer[] i = new Integer[array.size()];// 数组的长度和集合的长度一样

                   // Integer[] ii = array.toArray(i);

                   array.toArray(i);

                   // System.out.println(i);//i和ii其实是同一个

                   // System.out.println(ii);

 

                   // 对数组排序

                   // public static void sort(Object[] a)

                   Arrays.sort(i);

 

                   // 获取该数组中的最大索引的值

                   System.out.println("数组是：" + arrayToString(i) + "最大值是:" + i[i.length - 1]);

         }

 

         public static String arrayToString(Integer[] i)//遍历数组

         {

                   StringBuilder sb = new StringBuilder();

 

                   sb.append("[");

                   for (int x = 0; x < i.length; x++)

                   {

                            if (x == i.length - 1)

                            {

                                     sb.append(i[x]);

                            } else

                            {

                                     sb.append(i[x]).append(", ");

                            }

                   }

                   sb.append("]");

 

                   return sb.toString();

         }

}
