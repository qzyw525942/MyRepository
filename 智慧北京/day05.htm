// 继续发送延时3秒的信息，形成内循环

 

                                }

                        };

 

                        handler.sendEmptyMessageDelayed(0, 3000);

// 保证启动自动轮播逻辑只执行一次。发送延时3秒的消息

 

                        // 当用户触摸在vp上时，停止轮播

                        vp.setOnTouchListener(new OnTouchListener()

                        {

 

                                @Override

                                public boolean onTouch(View v, MotionEvent event)

                                {

 

                                        switch (event.getAction())

                                        {

                                        case MotionEvent.ACTION_DOWN:

 

                                                // 删除handler的消息

                                                handler.removeCallbacksAndMessages(null);

 

                                                break;

 

                                        case MotionEvent.ACTION_UP:

 

                                                handler.sendEmptyMessageDelayed(0, 3000);

 

                                                break;

 

                                        case MotionEvent.ACTION_CANCEL:

 

                                                handler.sendEmptyMessageDelayed(0, 3000);

 

                                                break;

 

                                        default:

                                                break;

                                        }

 

                                        return false;

                                }

                        });

 

                }

/**

 * 组图标签

public class ZuTuSlidingDetailPage extends BaseSlidingDetailPage implements

                OnClickListener

{

 

        private ListView lv;

        private GridView gv;

        private ArrayList<PhotoNews> newsData;

        private MyAdapter adapter;

 

        ImageButton ib_photo;

 

        public ZuTuSlidingDetailPage(Activity activity, ImageButton ib_photo)

        {

                super(activity);

 

                ib_photo.setOnClickListener(this);

 

                this.ib_photo = ib_photo;

 

        }

 

        @Override

        public View initView()

        {

                /**

                 * 第一步：找到各个控件

                 */

                View view = View.inflate(mActivity,

                                R.layout.zutuslidingdetailpage_lv_gv, null);

                lv = (ListView) view.findViewById(R.id.lv);

                gv = (GridView) view.findViewById(R.id.gv);

 

                return view;

        }

 

        @Override

        public void initData()

        {

                super.initData();

 

                /**

                 * 第四步：设置缓存

                 */

                String cache = CacheDataUtils.getCache(mActivity, GlobalVar.PHOTOS_URL);

                if (!TextUtils.isEmpty(cache))

                {

                        parserData(cache);

                }

 

                getDataFromNet();

 

        }

 

        /**

         * 第三步：访问网络解析数据，获得数据

         */

        private void getDataFromNet()

        {

                HttpUtils utils = new HttpUtils();

 

                utils.send(HttpMethod.GET, GlobalVar.PHOTOS_URL,

                                new RequestCallBack<String>()

                                {

 

                                        @Override

                                        public void onSuccess(ResponseInfo<String> responseInfo)

                                        {

                                                String result = responseInfo.result;

 

                                                CacheDataUtils.setCache(mActivity,

                                                                GlobalVar.PHOTOS_URL, result);

 

                                                parserData(result);

 

                                        }

 

                                        @Override

                                        public void onFailure(HttpException error, String msg)

                                        {

                                                error.printStackTrace();

 

                                                Toast.makeText(mActivity, msg, 1).show();

                                        }

                                });

 

        }

 

        private void parserData(String cache)

        {

                Gson gson = new Gson();

                PhotosBean jsonData = gson.fromJson(cache, PhotosBean.class);

 

                newsData = jsonData.data.news;

 

                /**

                 * 第七步：设置adapter给ListView

                 */

                lv.setAdapter(new MyAdapter());

                gv.setAdapter(new MyAdapter());

 

        }

 

        /**

         * 第二步：设置Adapter

         *

         */

        class MyAdapter extends BaseAdapter

        {

 

                private BitmapUtils bitmapUtils;

 

                public MyAdapter()

                {

                        super();

 

                        bitmapUtils = new BitmapUtils(mActivity);

 

                }

 

                @Override

                public int getCount()

                {

 

                        return newsData.size();

                }

 

                @Override

                public Object getItem(int position)

                {

                        return newsData.get(position);

                }

 

                @Override

                public long getItemId(int position)

                {

                        return position;

                }

 

                @Override

                public View getView(int position, View convertView, ViewGroup parent)

                {

                        /**

                         * 第五步：设置ListView的缓存

                         */

                        View view;

                        ViewHolder holder;

                        if (convertView == null)

                        {

                                view = View.inflate(mActivity,

                                                R.layout.zutuslidingdetailpage_lv_item, null);

                                holder = new ViewHolder();

 

                                holder.iv = (ImageView) view.findViewById(R.id.iv);

                                holder.title = (TextView) view.findViewById(R.id.title);

 

                                view.setTag(holder);

 

                        } else

                        {

                                view = convertView;

 

                                holder = (ViewHolder) view.getTag();

                        }

 

                        /**

                         * 第六步：设置数据给ListView条目的各个控件

                         */

                        PhotoNews photoNews = newsData.get(position);

                        holder.title.setText(photoNews.title);

 

                        // 拼接url

                        String imgUrl = photoNews.listimage;

                        String[] split = imgUrl.split("/");

                        String imgUrl0 = GlobalVar.BASE_URL + "/" + split[4] + "/"

                                        + split[5] + "/" + split[6];

 

                        LogUtils.i("地址", imgUrl0);

 

                        bitmapUtils.display(holder.iv, imgUrl0);

 

                        return view;

                }

        }

 

        static class ViewHolder

        {

                ImageView iv;

                TextView title;

        }

 

        /**

         * 第八步：设置标题栏右上角的ImageView的点击事件

         */

        public boolean isListView = true;

 

        @Override

        public void onClick(View v)

        {

                if (isListView)

                {

                        lv.setVisibility(View.VISIBLE);

                        lv.setVisibility(View.INVISIBLE);

 

                        ib_photo.setImageResource(R.drawable.icon_pic_list_type);

 

                        isListView = false;

 

                } else

                {

                        lv.setVisibility(View.INVISIBLE);

                        lv.setVisibility(View.VISIBLE);

 

                        ib_photo.setImageResource(R.drawable.icon_pic_grid_type);

 

                        isListView = true;

                }

        }
