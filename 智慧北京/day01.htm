第二集  SplashActivity的开发

 

新建智慧北京项目。在清单文件中更改项目的名称和图标，图标在今天的资源文件中有。以后凡是涉及资源文件的都在当天的资料中去找，不在赘述。新建splashActivity类。其界面布局文件如下：

界面分为红色的背景和ImageView的小马图片。

此界面在进入的时候具有旋转，渐变和动画的特效。

 

/**

 * SplashActivity页面

 */

public class SplashActivity extends Activity

{

 

         private RelativeLayout rlRoot;

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                   super.onCreate(savedInstanceState);

                   setContentView(R.layout.activity_splash);

 

                   rlRoot = (RelativeLayout) findViewById(R.id.rl_root);

                   // activity_splash的根节点

 

                   // 旋转动画

                   RotateAnimation animRotate = new RotateAnimation(0, 360, Animation.RELATIVE_TO_SELF, 0.5f,

                                     Animation.RELATIVE_TO_SELF, 0.5f);

                   animRotate.setDuration(1000);// 动画时间

                   animRotate.setFillAfter(true);// 保持动画结束状态

 

                   // 缩放动画

                   ScaleAnimation animScale = new ScaleAnimation(0, 1, 0, 1, Animation.RELATIVE_TO_SELF, 0.5f,

                                     Animation.RELATIVE_TO_SELF, 0.5f);

                   animScale.setDuration(1000);

                   animScale.setFillAfter(true);// 保持动画结束状态

 

                   // 渐变动画

                   AlphaAnimation animAlpha = new AlphaAnimation(0, 1);

                   animAlpha.setDuration(2000);// 动画时间

                   animAlpha.setFillAfter(true);// 保持动画结束状态

 

                   // 动画集合

                   AnimationSet set = new AnimationSet(true);

                   set.addAnimation(animRotate);

                   set.addAnimation(animScale);

                   set.addAnimation(animAlpha);

 

                   // 启动动画

                   rlRoot.startAnimation(set);

 

                   // 设置动画监听器

                   set.setAnimationListener(new AnimationListener()

                   {

 

                            @Override

                            public void onAnimationStart(Animation animation)

                            {

 

                            }

 

                            @Override

                            public void onAnimationRepeat(Animation animation)

                            {

 

                            }

 

                            // 动画结束,跳转页面：如果是第一次进入, 跳到新手引导页，否则跳主页面

                            @Override

                            public void onAnimationEnd(Animation animation)

                            {

 

                                     boolean isFirstEnter = PrefUtils.getBoolean(SplashActivity.this, "is_first_enter", true);

//PrefUtils见后

 

                                     Intent intent;

                                     if (isFirstEnter)

                                     {

                                               // 新手引导

                                               intent = new Intent(getApplicationContext(), GuideActivity.class);

 

                                     } else

                                     {

                                               // 主页面

                                               intent = new Intent(getApplicationContext(), MainActivity.class);

 

                                     }

 

                                     startActivity(intent);

 

                                     finish();// 结束当前页面

                            }

                   });

         }

 

}

 

/**

 * SharePreference封装

 *

 */

public class PrefUtils

{

 

         public static boolean getBoolean(Context ctx, String key, boolean defValue)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   return sp.getBoolean(key, defValue);

         }

 

         public static void setBoolean(Context ctx, String key, boolean value)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   sp.edit().putBoolean(key, value).commit();

         }

 

         public static void setString(Context ctx, String key, String value)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   sp.edit().putString(key, value).commit();

 

         }

 

         public static String getString(Context ctx, String key, String defValue)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   return sp.getString(key, defValue);

         }

 

         public static void setInt(Context ctx, String key, int value)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   sp.edit().putInt(key, value).commit();

         }

 

         public static int getInt(Context ctx, String key, int defValue)

         {

 

                   SharedPreferences sp = ctx.getSharedPreferences("config", Context.MODE_PRIVATE);

 

                   return sp.getInt(key, defValue);

         }

}

 

第三集  新手引导页的开发

 

从这里开始,分步来开发,这样逻辑会清晰些.

 

第一步：GuideActivity的界面布局文件如下：

三个控件都在相对布局中。注意相对布局是可以允许这个控件在另一个控件上面的。

 

GuideActivity如下：

//新手引导页

public class GuideActivity extends BaseActivity

{

 

         private ViewPager vp;

         private Button btn;

         private LinearLayout ll;

         private ImageView iv;

 

         private int pointDis;

 

         /**

          * 第三步：设置图片id数组，生成ImageView并存放到list中

          */

         int[] imgId = new int[] { R.drawable.guide_1, R.drawable.guide_2,

                            R.drawable.guide_3 };

 

         ArrayList<ImageView> list = new ArrayList<ImageView>();

 

         @Override

         protected void onCreate(Bundle savedInstanceState)

         {

                   super.onCreate(savedInstanceState);

                   setContentView(R.layout.activity_newcustom);

 

                   vp = (ViewPager) findViewById(R.id.vp);

                   btn = (Button) findViewById(R.id.btn);

                   ll = (LinearLayout) findViewById(R.id.ll);

 

                   /**

                    * 第六步：修改界面布局文件，在原来的几个小原点之上增加一个小红点。 底部的小圆点是不动的，小红点在它们之上左右浮动。原来的小圆点位于线性布局中，新加的小红点位于ImageView中，两个部分位于相对布局中。

                    */

                   iv = (ImageView) findViewById(R.id.iv);

 

                   initData();

 

                   vp.setAdapter(new MyAdapter());

 

                   /**

                    * 第七步：设置ViewPage滑动监听事件.从而获得小红点的平滑浮动位移值

                    */

                   vp.setOnPageChangeListener(new OnPageChangeListener()

                   {

 

                            @Override

                            public void onPageSelected(int position)

                            {

                                     /**

                                      * 第十步:设置开始体验按钮的显示隐藏.并为按钮设置状态选择器.以及点击事件

                                      */

                                     if (position == list.size() - 1)// 最后一页显示，其他页隐藏

                                     {

                                               btn.setVisibility(View.VISIBLE);

 

                                               // 跳转到主页面

                                               btn.setOnClickListener(new OnClickListener()

                                               {

 

                                                        @Override

                                                        public void onClick(View v)

                                                        {

                                                                 Intent intent = new Intent(GuideActivity.this,

                                                                                    MainActivity.class);

                                                                 startActivity(intent);

 

                                                                 // 改变sp中的值

                                                                 SPUtil.setBoolean(getApplicationContext(),

                                                                                    "is_first_enter", false);

 

                                                                 // 关闭新手引导页

                                                                 finish();

 

                                                        }

                                               });

 

                                     } else

                                     {

                                               btn.setVisibility(View.INVISIBLE);

                                     }

                            }

 

                            @Override

                            public void onPageScrolled(int position, float positionOffset,

                                               int positionOffsetPixels)

                            {

                                     System.out.println("当前位置：" + position + "；移动时两个页之间的移动百分比："

                                                        + positionOffset);

 

                                     /**

                                      * 第九步：更新小红点的位置。小红点的位置=两个小圆点之间的距离pointDis*移动时两个页（即小圆点）

                                      * 之间的移动百分比positionOffset。这个值实际上也就是小红点距离左边的左边距

                                      */

                                     int marginLeft = (int) (pointDis * positionOffset) + position

                                                        * pointDis;

                                     // 后面之所以要加上position* pointDis，是因为要记住之前滑动到的位置。position的取值为0 1

                                     // 2...,代表从左到右的小圆点顺序

 

                                     // 小红点的父控件是RelativeLayout

                                     RelativeLayout.LayoutParams params = (LayoutParams) iv

                                                        .getLayoutParams();

                                     params.leftMargin = marginLeft;

                                     iv.setLayoutParams(params);

 

                            }

 

                            @Override

                            public void onPageScrollStateChanged(int state)

                            {

 

                            }

                   });

 

                   /**

                    * 第八步：监听onLayout方法什么时候执行完。只有执行完才能得到控件之间的距离

                    * 注意：两个小圆点之间的距离pointDis不是第五步中设置的params.leftMargin = 10;

                    * 因为不同屏幕的这个值是不一样的。

                    */

                   iv.getViewTreeObserver().addOnGlobalLayoutListener(

                                     new OnGlobalLayoutListener()

                                     {

 

                                               @Override

                                               public void onGlobalLayout()

                                               {

                                                        // android系统底层会多次调用此方法。为了只能调用一次此方法，所以如下：

                                                        iv.getViewTreeObserver().removeOnGlobalLayoutListener(

                                                                           this);

                                                        // 这里的this代表OnGlobalLayoutListener

 

                                                        // onLayout方法执行完成之后调用这里。在这里获取两个小圆点之间的距离。

                                                        // 只有先运行onLayout方法才能得到控件之间的距离、

                                                        // 移动的距离=第二个圆点的左边距-第一个圆点的左边距

                                                        pointDis = ll.getChildAt(1).getLeft()

                                                                           - ll.getChildAt(0).getLeft();

 

                                               }

                                     });

         }

 

         public void initData()

         {

 

                   for (int i = 0; i < imgId.length; i++)

                   {

                            ImageView imageView = new ImageView(this);

                            imageView.setBackgroundResource(imgId[i]);// 图片自动填充ImageView

                            // imageView.setImageResource(imgId[i]);// 图片不自动填充ImageView

 

                            list.add(imageView);

 

                            /**

                             * 第五步：初始化小圆点

                             */

                            ImageView point = new ImageView(this);

                            point.setImageResource(R.drawable.shape_point);

                            // shape_point.xml如下：

                            /*

                             * <?xml version="1.0" encoding="utf-8"?> <shape

                             * xmlns:android="http://schemas.android.com/apk/res/android"

                             * android:shape="oval" >

                             *

                             * <solid android:color="#cccccc" />

                             *

                             * <size android:height="10dp" android:width="10dp" />

                             *

                             * </shape>

                             */

 

                            // 小圆点的父控件是LinearLayout

                            LinearLayout.LayoutParams params = new LinearLayout.LayoutParams(

                                               LinearLayout.LayoutParams.WRAP_CONTENT,

                                               LinearLayout.LayoutParams.WRAP_CONTENT);

 

                            if (i > 0)

                            {

                                     // 从第二个小圆点开始，距离左边小圆点10dp

                                     params.leftMargin = 10;

                            }

 

                            ll.addView(point, params);

 

                   }

 

         }

 

         /**

          * 第二步：找到所有控件并设置ViewPager的Adapter

          */

         public class MyAdapter extends PagerAdapter

         {

 

                   @Override

                   public int getCount()

                   {

                            return list.size();

                   }

 

                   // 是否缓存复用:固定格式

                   @Override

                   public boolean isViewFromObject(View view, Object object)

                   {

                            return view == object;

                   }

 

                   @Override

                   public Object instantiateItem(ViewGroup container, int position)

                   {

                            /**

                             * 第四步：设置getCount方法的返回值并把ImageView设置给ViewPage和框架

                             */

 

                            ImageView imageView = list.get(position);

 

                            container.addView(imageView);

 

                            return imageView;

                   }

 

                   // 删除条目：固定格式

                   @Override

                   public void destroyItem(ViewGroup container, int position, Object object)

                   {

                            container.removeView((View) object);

                   }

 

         }

}

 

 

 

 

第四集  MainActivity开发——SlidingMenu

 

库文件和jar包都能导入到项目中来。不同的是库文件中java代码和资源文件都能导入项目，用法与在本项目中的一样；而jar包只能导入java代码，用法与在本项目中的一样。

/**

/**

 *

 * 第二步：MainActivity继承自SlidingFragmentActivity

public class MainActivity extends SlidingFragmentActivity

{

 

         /**

          * 第三步：修改onCreate方法为public

          */

         @Override

         public void onCreate(Bundle savedInstanceState)

         {

                   super.onCreate(savedInstanceState);

                   requestWindowFeature(Window.FEATURE_NO_TITLE);

                   setContentView(R.layout.activity_main);

 

                   /**

                    * 第四步：加载左侧界面布局

                    */

                   setBehindContentView(R.layout.activity_left);

 

                   /**

                    * 第五步：加载右侧界面布局

                    */

                   SlidingMenu menu = getSlidingMenu();

                   menu.setSecondaryMenu(R.layout.activity_right);

 

                   /**

                    * 第六步：调用相关API设置

                    */

                   // 设置左右菜单都能显示

                   menu.setMode(SlidingMenu.LEFT_RIGHT);

                   // 设置在屏幕任何一个地方都能拉出左右菜单，而不是只能在左边沿或者右边沿才能拉出

                   menu.setTouchModeAbove(SlidingMenu.TOUCHMODE_FULLSCREEN);

                   // 设置左右菜单拉开以后，留给主页面的宽度,这样就不至于拉开以后主页面就不见了

                   menu.setBehindOffset(100);

 

         }

}
