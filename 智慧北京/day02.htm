public abstract class BaseFragment extends Fragment

{

        private Activity mActivity;

 

        // Fragment创建

        @Override

        public void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

 

                /**

                 * 第二步： 在onCreate方法中获取Fragment所依赖的Activity

                 */

                mActivity = getActivity();

 

        }

 

        // 初始化Fragment的界面布局

        @Override

        public View onCreateView(LayoutInflater inflater, ViewGroup container,

                        Bundle savedInstanceState)

        {

 

                /**

                 * 第三步：每个子类Fragment的界面都不一样，所以BaseFragment无法具体指定。

                 * 但是父类可以提供抽象方法让子类去实现，父类再调用此抽象方法

                 */

                View view = initView();

 

                return view;

        }

 

        public abstract View initView();

 

        // Fragment所依赖的Activity的onCreate方法执行完毕

        @Override

        public void onActivityCreated(Bundle savedInstanceState)

        {

                super.onActivityCreated(savedInstanceState);

 

                /**

                 * 第四步： 由子类去初始化数据

                 */

                initData();

        }

 

        public abstract void initData();

}

public class MainActivity extends SlidingFragmentActivity

{

        public static final String TAG_MAIN = "TAG_MAIN";

        public static final String TAG_Left_Menu = "TAG_Left_Menu";

 

        @Override

        public void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                requestWindowFeature(Window.FEATURE_NO_TITLE);

                setContentView(R.layout.activity_main);

 

                setBehindContentView(R.layout.activity_main_left);

                SlidingMenu menu = getSlidingMenu();

                menu.setTouchModeAbove(SlidingMenu.TOUCHMODE_FULLSCREEN);

                menu.setBehindOffset(100);

 

                initFragment();

        }

 

        /**

         * 第四步：用Fragment替换原有的界面布局

         */

        public void initFragment()

        {

                FragmentManager fm = getSupportFragmentManager();

 

                FragmentTransaction ft = fm.beginTransaction();

 

                // 第一个参数是原有的界面布局，第二个参数是要替换的Fragment，第三个参数是为每个Fragment做一个标记，方便后面寻找

                ft.replace(R.id.fl_left, new LeftMenuFragment(), TAG_Left_Menu);

                ft.replace(R.id.fl_main, new MainFragment(), TAG_MAIN);

 

                ft.commit();

 

        }

}

// MainActivity的主侧边栏

public class MainFragment extends BaseFragment

{

 

        private ViewPager vp;

        private ArrayList<BasePage> pageList;

 

        @Override

        public View initView()

        {

                View view = View.inflate(mActivity, R.layout.fragment_main_main, null);

 

                /**

                 * 第一步：找到Vp

                 */

                vp = (ViewPager) view.findViewById(R.id.vp);

 

                return view;

        }

 

        @Override

        public void initData()

        {

 

                /**

                 * 第三步：创建五个标签页的集合，并存入五个标签页

                 */

                pageList = new ArrayList<BasePage>();

 

                pageList.add(new HomePage(mActivity));

                pageList.add(new NewsPage(mActivity));

                pageList.add(new ZhihuiPage(mActivity));

                pageList.add(new GovPage(mActivity));

                pageList.add(new SettingPage(mActivity));

 

                LogUtils.i("MainFragment的mActivity:", mActivity + "");

 

                /**

                 * 第五步：设置MyAdapter给vp

                 */

                vp.setAdapter(new MyAdapter());

 

        }

 

        /**

         * 第二步:设置vp的adapter,并完成isViewFromObject和destroyItem方法

         */

        class MyAdapter extends PagerAdapter

        {

 

                @Override

                public int getCount()

                {

                        return pageList.size();

                }

 

                @Override

                public boolean isViewFromObject(View view, Object object)

                {

                        return view == object;

                }

 

                @Override

                public Object instantiateItem(ViewGroup container, int position)

                {

                        /**

                         * 第四步：完成getCount和instantiateItem方法

                         */

                        BasePage basePage = pageList.get(position);

                        View view = basePage.rootView;

 

                        // 这里需要调用此方法，否则其它地方无人调用，将无法初始化数据

                        basePage.initData();

 

                        container.addView(view);

                        return view;

                }

 

                @Override

                public void destroyItem(ViewGroup container, int position, Object object)

                {

 

                        container.removeView((View) object);

                }

        }

 

}
