第四步：书写c代码：

#include <jni.h>

#include <stdlib.h>

 

/*

 * 加密方法：

 *

 * Class:     com_example_word_MainActivity

 * Method:    encodeInC

 * Signature: (Ljava/lang/String;)Ljava/lang/String;

 */

JNIEXPORT jstring JNICALL Java_com_example_word_MainActivity_encodeInC(

                JNIEnv * env, jobject this, jstring jstr)

{

        //把java传递进来的String转为char*

        char* cstr = _JString2CStr(jstr);

 

        //获取cstr的长度

        int length = strlen(cstr);

 

        //遍历

        int i;

        for (i = 0; i < length; i++)

        {

 

                *(cstr + i) += 2; //加密处理：字符数组的每一个字符元素加2

 

        }

 

        //再把char*转化为String并返回

        return (*env)->NewStringUTF(env, cstr);

 

}

 

/*

 * 解密方法：

 *

 * Class:     com_example_word_MainActivity

 * Method:    decodeInC

 * Signature: (Ljava/lang/String;)Ljava/lang/String;

 */

JNIEXPORT jstring JNICALL Java_com_example_word_MainActivity_decodeInC(

                JNIEnv * env, jobject this, jstring jstr)

{

 

        //把java传递进来的String转为char*

        char* cstr = _JString2CStr(jstr);

 

        //获取cstr的长度

        int length = strlen(cstr);

 

        //遍历

        int i;

        for (i = 0; i < length; i++)

        {

 

                *(cstr + i) -= 2; //加密处理：字符数组的每一个字符元素加2

 

        }

 

        //再把char*转化为String并返回

        return (*env)->NewStringUTF(env, cstr);

 

}

 

 

/**

 * 把java传递过来的String转换成c语言的char* 类型.

* 实际开发中应该把此函数放在调用的函数之前，否则编译会出错

 */

char* _JString2CStr(JNIEnv* env, jstring jstr)

{

 

        char* rtn = NULL;

        // NULL：没有意义

 

        jclass clsstring = (*env)->FindClass(env, "java/lang/String");

        //找到String的字节码对象

 

        jstring strencode = (*env)->NewStringUTF(env, "GB2312");

        //把char*类型的GB2312转化为jstring类型

 

        jmethodID mid = (*env)->GetMethodID(env, clsstring, "getBytes",

                        "(Ljava/lang/String;)[B");

        //获得String中的参数为String，返回值为byte数组([B之义)的getBytes方法

 

        jbyteArray barr = (jbyteArray) (*env)->CallObjectMethod(env, jstr, mid,

                        strencode);

        //相当于：String .getByte("GB2312");

 

        jsize alen = (*env)->GetArrayLength(env, barr);

        //获得barr的长度

 

        jbyte* ba = (*env)->GetByteArrayElements(env, barr, JNI_FALSE);

        //拿到字节数组元素的首地址

 

        if (alen > 0)

        {

 

                rtn = (char*) malloc(alen + 1);

                //申请内存地址，加1用于存放："\0"

 

                memcpy(rtn, ba, alen);

                //数组拷贝。把字节数组拷贝到rtn申请的内存地址中

 

                rtn[alen] = 0;

                //把"\0"放入数组

        }

 

        (*env)->ReleaseByteArrayElements(env, barr, ba, 0);

        //释放字节数组元素

 

        return rtn;

}

第二集  黑马秀秀案例

在今天的资料中有美图秀秀apk；反编译此美图秀秀后,可见有一个JNI的类和.so文件。如果知道JNI类中native方法的参数含义；再根据这两个文件，就能使用.so文件中的东西了。因为JNI声明的方法在c代码中是一一对应的，而c代码又编译成.so文件，所以JNI声明的方法在.so中也是一一对应的。

 

下面根据美图调用其JNI中的AutoWhiteBalance方法的使用方式来弄清这个方法中各个参数的含义；再根据JNI类和.so两个文件，在自己的项目中使用这个方法。

 

新建一个项目。复制美图的.so文件到项目的如下路径中：

 

复制美图的JNI类到项目中。因为JNI类中的native方法对应在c函数声明中是与包名有关联的；所以要在自己项目中新建一个与美图JNI原有包名一致的包，再把美图的JNI类复制进去。并在其中增加如下：

项目的界面布局文件如下，按钮下方是一个ImageView，java代码中指定其图片。

 

MainActivity如下：

public class MainActivity extends Activity

{

        private ImageView image;

        private Bitmap bm;

        private JNI jni;

 

        @Override

        protected void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                setContentView(R.layout.activity_main);

 

                image = (ImageView) findViewById(R.id.image);

 

                bm = BitmapFactory.decodeFile("sdcard/awaiyi.jpg");

 

                image.setImageBitmap(bm);

 

                jni = new JNI();

        }

 

        // 按钮的点击事件

        public void process(View v)

        {

                int width = bm.getWidth();

                // 宽度表示图片一行有多少个像素

 

                int height = bm.getHeight();

                // 宽度表示图片一列有多少个像素

 

                // 一张图片可以转化成一个数组：

 

// 创建一个数组，用于存放图像转化成的数据。一个像素对应于一个int

//数据

                int[] pixels = new int[width * height];

 

                // 第二个参数表示偏移量，设为0

                // 第三个参数表示逐行扫描，

                // 后面参数表示从（0,0）位置开始，到（width, height）结束

                bm.getPixels(pixels, 0, width, 0, 0, width, height);

                // getpixels运行后 pixels数组就被修改了。pixels中保存了bm的颜色信息

 

// StyleBaoColor运行后 pixels的颜色信息已经被修改。修改成特效处理//后的颜色信息

                jni.StyleBaoColor(pixels, width, height);

                // 查看反编译的JNI文件可见，StyleBaoColor方法的参数和

//AutoWhiteBalance参数是一样的

 

                // 用处理好的像素数组创建一张新的图片 这张图片就是经过特效处理的

                Bitmap bm2 = Bitmap.createBitmap(pixels, width, height, bm.getConfig());

 

                image.setImageBitmap(bm2);

        }

}

 

第三集  压力表ProgressBar案例

 

MainActivity如下：

public class MainActivity extends Activity

{

        static

        {

                System.loadLibrary("YaLiTEST");

        }

 

        ProgressBar pb;

 

        @Override

        protected void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                setContentView(R.layout.activity_main);

 

                pb = (ProgressBar) findViewById(R.id.pb);

 

        }

 

        // 开始检测按钮

        public void start(View v)

        {

                new Thread()

                {

                        public void run()

                        {

                                startMoniter();

                        };

                }.start();

 

        }

 

        // 停止检测按钮

        public void stop(View v)

        {

                stopMoniter();

        }

 

        // 开始检测

        public native void startMoniter();

 

        // 停止检测

        public native void stopMoniter();

 

        // 调用startMoniter方法转入c代码，再调用此方法设置进度值

        public void setPressure(int pressure)

        {

                pb.setProgress(pressure);

        }

}

 

C代码如下：

#include <jni.h>

#include <stdlib.h>

 

//这里使用此方法模拟获得硬件检测到的值。

int getPressure()

{

        return rand() % 100;

}

 

int flag = 0;

 

JNIEXPORT void JNICALL Java_com_example_yalitest_MainActivity_startMoniter(

                JNIEnv * env, jobject obj)

{

        flag = 1;

 

        while (flag)

        {

                sleep(1);

 

                //1、拿到字节码对象

                jclass clazz = (*env)->FindClass(env,"com/example/yalitest/MainActivity");

 

                //2、获得方法

                jmethodID methodID = (*env)->GetMethodID(env, clazz, "setPressure","(I)V");

 

                //3、调用setPressure方法

                (*env)->CallVoidMethod(env, obj, methodID, getPressure());

        }

 

}

 

 

JNIEXPORT void JNICALL Java_com_example_yalitest_MainActivity_stopMoniter(

                JNIEnv * env, jobject obj)

{

        flag = 0;

}

/**

public class MyProgressBar extends View

{

        int pressure = 0;

 

        Paint paint;

 

        public MyProgressBar(Context context, AttributeSet attrs, int defStyle)

        {

                super(context, attrs, defStyle);

 

                init();

        }

 

        public MyProgressBar(Context context, AttributeSet attrs)

        {

                super(context, attrs);

 

                init();

        }

 

        public MyProgressBar(Context context)

        {

                super(context);

 

                init();

        }

 

        /**

         * 初始化方法

         */

        public void init()

        {

                // 新建一支画笔

                paint = new Paint();

        }

 

        @Override

        protected void onDraw(Canvas canvas)

        {

                super.onDraw(canvas);

 

                if (pressure < 40)

                {

                        paint.setColor(Color.BLACK);

 

                } else if (pressure < 80)

                {

                        paint.setColor(Color.BLUE);

 

                } else

                {

                        paint.setColor(Color.RED);

 

                }

 

                // 开始的四个参数分别是长方形left, top, right, bottom距离屏幕左边，顶部，左边，顶部的距离，见下图：       

 

 

 

                 

                // 要改变的是长方形的高低，所以底部不动，高度变化

                canvas.drawRect(50, 200 - pressure, 100, 200, paint);

 

                // 中间的两个参数分别是字体相对于自定义控件左边和顶部的距离

                canvas.drawText("当前压力值" + pressure, 50, 50, paint);

 

                // 设置字体大小

                paint.setTextSize(45);

        }

 

        // 设置pressure值的方法

        public void setPressure(int pressure)

        {

                this.pressure = pressure;

 

                // 重绘视图

                // invalidate只能在主线程调用，子线程用postInvalidate()

                // invalidate();

                postInvalidate();

        }

}

C代码如下：

#include <jni.h>

//用<>：直接到系统指定的include目录下去找

 

#include "com_itheima_cppJNI_MainActivity.h"

//用""： 系统会先到c源代码所在的文件夹去找头文件 ,如果找不到再到系统指定的incude文件夹下找。所以生成的.h不能删除，要剪切到.cpp文件所在的目录。

 

JNIEXPORT jstring JNICALL Java_com_itheima_cppJNI_MainActivity_helloFromCPP(

                JNIEnv * env, jobject obj)

{

        return env->NewStringUTF("hello from cpp");

        //注意在c++中，env是结构体的一级指针，不再是二级指针，所以不再用(*env)的形式

}
