public class MainActivity extends Activity

{

 

        private EditText mEtName;

        private EditText mEtUID;

        private EditText mEtPwd;

 

        @Override

        protected void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                setContentView(R.layout.activity_main);

 

                initView();

        }

 

        private void initView()

        {

                mEtName = (EditText) findViewById(R.id.et_name);

                mEtUID = (EditText) findViewById(R.id.et_uid);

                mEtPwd = (EditText) findViewById(R.id.et_pwd);

        }

 

        public void startCon(View v)

        {

                String name = mEtName.getText().toString().trim();

                String uid = mEtUID.getText().toString().trim();

                String pwd = mEtPwd.getText().toString().trim();

 

                if (TextUtils.isEmpty(name))

                {

                        Toast.makeText(getApplicationContext(), "名称不能为空",

                                        Toast.LENGTH_SHORT).show();

 

                } else if (TextUtils.isEmpty(uid) && uid.length() != 20)

                {

                        Toast.makeText(getApplicationContext(), "UID错误", Toast.LENGTH_SHORT)

                                        .show();

 

                } else if (TextUtils.isEmpty(pwd))

                {

                        Toast.makeText(getApplicationContext(), "密码不能为空",

                                        Toast.LENGTH_SHORT).show();

 

                } else

                {

                        Intent intent = new Intent(this, CameraActivity.class);

                        intent.putExtra("name", name);

                        intent.putExtra("uid", uid);

                        intent.putExtra("pwd", pwd);

                        startActivity(intent);

                        finish();

                }

 

        }

}

public class CameraActivity extends Activity

{

 

        private Monitor mMonitor;

        private TextView mTvStatus;

        private Camera mCamera;

 

        private MyIotcListener mMyIotcListener;

 

        /**

         * 第六步：Handler接受数据更新界面

         */

        private Handler mHandler = new Handler()

        {

                public void handleMessage(android.os.Message msg)

                {

                        switch (msg.what)

                        {

 

                        case Camera.CONNECTION_STATE_CONNECTING:

                                // 连接中

                                mTvStatus.setText("连接中");

                                break;

 

                        case Camera.CONNECTION_STATE_CONNECTED:

                                // 连接成功

                                mTvStatus.setText("连接成功");

 

                                // 显示摄像头画面

                                showCamera();

 

                                break;

 

                        case Camera.CONNECTION_STATE_CONNECT_FAILED:

                                // 连接失败

                                mTvStatus.setText("连接失败");

                                break;

 

                        case Camera.CONNECTION_STATE_TIMEOUT:

                                // 超时

                                mTvStatus.setText("连接超时");

                                break;

 

                        default:

                                break;

                        }

                };

        };

 

        @Override

        protected void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                setContentView(R.layout.activity_camera);

 

                initView();

 

                initValue();

 

        }

 

        /**

         * 第一步：找到控件

         */

        private void initView()

        {

                mMonitor = (Monitor) findViewById(R.id.monitor);

                mTvStatus = (TextView) findViewById(R.id.tv_status);

                // 显示连接状态的TextView

        }

 

        /**

         * 第二步：获得数据

         */

        private void initValue()

        {

                Intent intent = getIntent();

                String name = intent.getStringExtra("name");

                String uid = intent.getStringExtra("uid");

                String pwd = intent.getStringExtra("pwd");

                conn(name, uid, pwd);

        }

 

        /**

         * 第三步：连接摄像头

         */

        private void conn(String name, String uid, String pwd)

        {

                // 初始化摄像头, 初始化AV通道

                Camera.init();

 

                // 创建Camera实例对象

                mCamera = new Camera();

 

                /**

                 * 第四步：注册回调接口, 摄像头返回的信息通过回调方法传递给我们，位于Camera连接之前调用

                 */

                mMyIotcListener = new MyIotcListener();

                mCamera.registerIOTCListener(mMyIotcListener);

 

                // 连接摄像头, 参数为uid

                mCamera.connect(uid);

 

                // 开启摄像头

                // 参数1: 通道号, 后面还会用到, 要保证一致

                mCamera.start(Camera.DEFAULT_AV_CHANNEL, name, pwd);

 

                // 发送测试指令, 测试是否连接成功

                // 参数1: 通道号

                // 参数2: 指令类型

                // 参数3: 指令参数

                mCamera.sendIOCtrl(Camera.DEFAULT_AV_CHANNEL,

                                AVIOCTRLDEFs.IOTYPE_USER_IPCAM_GETSUPPORTSTREAM_REQ,

                                AVIOCTRLDEFs.SMsgAVIoctrlGetSupportStreamReq.parseContent());

 

        }

 

        /**

         * 第八步：点击按钮, 发送不同指令，控件摄像头旋转

         */

        public void sendCtrl(View v)

        {

                switch (v.getId())

                {

                case R.id.ib_left:

                        sendPtz(AVIOCTRLDEFs.AVIOCTRL_PTZ_LEFT);

                        break;

 

                case R.id.ib_right:

                        sendPtz(AVIOCTRLDEFs.AVIOCTRL_PTZ_RIGHT);

                        break;

 

                case R.id.ib_top:

                        sendPtz(AVIOCTRLDEFs.AVIOCTRL_PTZ_UP);

                        break;

 

                case R.id.ib_bottom:

                        sendPtz(AVIOCTRLDEFs.AVIOCTRL_PTZ_DOWN);

                        break;

 

                default:

                        break;

                }

        }

 

        private void sendPtz(int ptz)

        {

                byte b = (byte) ptz;

 

                mCamera.sendIOCtrl(Camera.DEFAULT_AV_CHANNEL,

 

                AVIOCTRLDEFs.IOTYPE_USER_IPCAM_PTZ_COMMAND,

 

                // 方向, 速度, 距离, 触摸点个数, 外接设备, 通道号

                                AVIOCTRLDEFs.SMsgAVIoctrlPtzCmd.parseContent(b, (byte) 0,

                                                (byte) 0, (byte) 0, (byte) 0,

                                                (byte) Camera.DEFAULT_AV_CHANNEL));

        }

 

        /**

         * 第九步：释放资源，断开连接。如果不释放的话，退出当前Activity再进入时无法连接摄像头了

         */

        @Override

        protected void onDestroy()

        {

                super.onDestroy();

 

                // 释放资源, 断开连接。将上面的方法反着调用一遍

                mCamera.stopShow(Camera.DEFAULT_AV_CHANNEL);

                mMonitor.deattachCamera();

                mCamera.stop(Camera.DEFAULT_AV_CHANNEL);

                mCamera.disconnect();

                mCamera.unregisterIOTCListener(mMyIotcListener);

                Camera.uninit();

        }

 

        /**

         * 第五步：设置回调接口

         */

        class MyIotcListener implements IRegisterIOTCListener

        {

 

                // 返回通道的信息

                @Override

                public void receiveChannelInfo(Camera arg0, int channel,

                                int responseCode)

                {

                        System.out.println("conn status: " + responseCode);

                        // 注意这里的responseCode不是http中的responseCode，这里的是Camera自定义的

 

                        // 这个回调方法是在子线程里调用的，所以要把结果发到主线程去更新界面

                        mHandler.sendEmptyMessage(responseCode);

                }

 

                // 返回摄像头捕捉到的画面信息

                @Override

                public void receiveFrameData(Camera arg0, int arg1, Bitmap arg2)

                {

 

                }

 

                // 返回摄像头其他信息

                @Override

                public void receiveFrameInfo(Camera arg0, int arg1, long arg2,

                                int arg3, int arg4, int arg5, int arg6)

                {

 

                }

 

                // 返回指令结过信息

                @Override

                public void receiveIOCtrlData(Camera arg0, int arg1, int arg2,

                                byte[] arg3)

                {

 

                }

 

                // 返回当前会话信息

                @Override

                public void receiveSessionInfo(Camera arg0, int arg1)

                {

 

                }

 

        }

 

        /**

         * 第七步： 显示画面, 需要通过 Monitor控件显示, 但数据从 Camera中获取

         */

        protected void showCamera()

        {

                // 判断是否连击成功

                if (mCamera != null

                                && mCamera.isChannelConnected(Camera.DEFAULT_AV_CHANNEL))

                {

                        // 设置最大焦距

                        mMonitor.setMaxZoom(1.0f);

 

                        // 挂载摄像头

                        mMonitor.attachCamera(mCamera, Camera.DEFAULT_AV_CHANNEL);

 

                        // 开始显示

                        mCamera.startShow(Camera.DEFAULT_AV_CHANNEL, true);

                }

        }

 

}

界面布局文件如下：

 

 

public class MainActivity extends Activity implements OnItemClickListener

{

 

        private ListView mLv;

        private BluetoothAdapter mBluetoothAdapter;

        private OutputStream mOutputStream;

        private DeviceAdapter mAdapter;

 

        private ArrayList<BluetoothDevice> mDevices = new ArrayList<BluetoothDevice>();

 

        /**

         * 第三步：设置获得扫描结果的BroadcastReceiver

         */

        private BroadcastReceiver mBluetoothReceiver = new BroadcastReceiver()

        {

                @Override

                public void onReceive(Context context, Intent intent)

                {

 

                        String action = intent.getAction();

 

                        if (BluetoothDevice.ACTION_FOUND.equals(action))

                        {

                                // 扫描到蓝牙设备。Parcelable：打包的

                                BluetoothDevice device = intent

                                                .getParcelableExtra(BluetoothDevice.EXTRA_DEVICE);

 

                                System.out.println("device name" + device.getName());

 

                                // 每扫描到一个设备，就往集合中去添加，并更新界面

                                mDevices.add(device);

                                mAdapter.notifyDataSetChanged();

 

                        } else if (BluetoothAdapter.ACTION_DISCOVERY_STARTED.equals(action))

                        {

                                Toast.makeText(getApplicationContext(), "开始扫描",

                                                Toast.LENGTH_SHORT).show();

 

                        } else if (BluetoothAdapter.ACTION_DISCOVERY_FINISHED

                                        .equals(action))

                        {

                                Toast.makeText(getApplicationContext(), "扫描结束",

                                                Toast.LENGTH_SHORT).show();

                        }

                }

        };

 

        @Override

        protected void onCreate(Bundle savedInstanceState)

        {

                super.onCreate(savedInstanceState);

                setContentView(R.layout.activity_main);

 

                /**

                 * 第七步：设置lv显示扫描到的设备

                 */

                mLv = (ListView) findViewById(R.id.lv);

                mAdapter = new DeviceAdapter(getApplicationContext(), mDevices);

                // DeviceAdapter类见后，为第八步

                mLv.setAdapter(mAdapter);

                mLv.setOnItemClickListener(this);

 

                mBluetoothAdapter = BluetoothAdapter.getDefaultAdapter();

 

                /**

                 * 第四步：注册广播接收者。当扫描到蓝牙设备的时候, 系统会发送广播；这里注册的广播接收者就能收到这些广播了

                 */

                IntentFilter filter = new IntentFilter();

                filter.addAction(BluetoothDevice.ACTION_FOUND);

                filter.addAction(BluetoothAdapter.ACTION_DISCOVERY_STARTED);

                filter.addAction(BluetoothAdapter.ACTION_DISCOVERY_FINISHED);

                registerReceiver(mBluetoothReceiver, filter);

        }

 

        /**

         * 第五步：记得一定要注销广播接受者

         */

        @Override

        protected void onDestroy()

        {

                super.onDestroy();

 

                unregisterReceiver(mBluetoothReceiver);

        }

 

        /**

         * 第一步：设置几个按钮的点击事件：

         */

        public void clickBtn(View v)

        {

                switch (v.getId())

                {

 

                /**

                 * 第二步：，获取蓝牙适配器；打开关闭蓝牙

                 */

                case R.id.button1:

 

                        // 判断蓝牙是否可用

                        if (!mBluetoothAdapter.isEnabled())

                        {

                                // 打开蓝牙

                                mBluetoothAdapter.enable();

                        }

 

                        break;

 

                case R.id.button2:

 

                        // 关闭蓝牙

                        if (mBluetoothAdapter.isEnabled())

                        {

                                mBluetoothAdapter.disable();

                        }

 

                        break;

 

                /**

                 * 第六步：扫描设备

                 */

                case R.id.button3:

 

                        // 开始扫描。扫描前清空数据集合

                        mDevices.clear();

                        mAdapter.notifyDataSetChanged();

 

                        mBluetoothAdapter.startDiscovery();

 

                        break;

 

                case R.id.button4:

 

                        // 停止扫描

                        mBluetoothAdapter.cancelDiscovery();

 

                        break;

 

                /**

                 * 第十一步：发送指令给灯泡，以控制其开，关，或闪亮的情况

                 */

                case R.id.button5:

                        sendCtrl(0);

                        break;

 

                case R.id.button6:

                        sendCtrl(1);

                        break;

 

                case R.id.button7:

                        sendCtrl(2);

                        break;

 

                default:

                        break;

                }

        }

 

        /**

         * 第十二步：对指令进行拆解，转换成灯泡能理解的数据，再发送过去

         */

        private void sendCtrl(int i)

        {

                try

                {

// 查看蓝牙灯泡的指令表可见，灯泡开，关，或闪亮的情况由三//条指令数据控制，如下；

                        // 打开： 01 99 10 10 99

                        // 关闭 ：01 99 11 11 99

                        // 闪亮：01 99 17 17 99

                        // 这些数据都是16进制；且前两个数据和最后一个数据都是一样的；又因为OutputStream的write方法接受的是byte类型的数据，

                        // 所以使用字节数组来容纳这些数据

                        byte[] bs = new byte[5];

                        bs[0] = (byte) 0x01;// 数组第一个元素

                        bs[1] = (byte) 0x99;// 数组第二个元素

 

                        // 数组第三、四个元素

                        if (i == 0)

                        {

                                bs[2] = (byte) 0x10;

                                bs[3] = (byte) 0x10;

 

                        } else if (i == 1)

                        {

                                bs[2] = (byte) 0x11;

                                bs[3] = (byte) 0x11;

 

                        } else if (i == 2)

                        {

                                bs[2] = (byte) 0x17;

                                bs[3] = (byte) 0x17;

 

                        }

 

                        // 数组第五个元素

                        bs[4] = (byte) 0x99;

 

                        mOutputStream.write(bs);

 

                } catch (IOException e)

                {

                        e.printStackTrace();

                }

        }

 

        /**

         * 第九步：设置lv的点击事件

         */

        @Override

        public void onItemClick(AdapterView<?> parent, View view, int position,

                        long id)

        {

                BluetoothDevice device = mDevices.get(position);

                conn(device);

        }

 

        /**

         * 第十步：连接设备

         */

        private void conn(final BluetoothDevice device)

        {

                // 建立蓝牙连接是耗时操作, 类似TCP Socket, 需要放在子线程里

                new Thread()

                {

                        public void run()

                        {

                                try

                                {

                                        // 因为是手机主动连接蓝牙灯泡，所以手机是客户端，灯泡是服务端

                                        // 获取 BluetoothSocket,

                                        // UUID需要和蓝牙灯泡服务端保持一致。其中这里的蓝牙灯泡使用的UUID是广泛使用的UUID,

                                        // 而这个UUID在createRfcommSocketToServiceRecord方法的API说明中有

                                        // 鼠标长停在此方法上，就有API说明，就能看到这个值了

                                        BluetoothSocket bluetoothSocket = device

                                                        .createRfcommSocketToServiceRecord(UUID

                                                                        .fromString("00001101-0000-1000-8000-00805F9B34FB"));

 

                                        // 和蓝牙服务端建立连接

                                        bluetoothSocket.connect();

 

                                        // 获取输出流, 往蓝牙服务端写指令信息

                                        mOutputStream = bluetoothSocket.getOutputStream();

 

                                        // 提示用户

                                        runOnUiThread(new Runnable()

                                        {

                                                public void run()

                                                {

 

                                                        Toast.makeText(getApplicationContext(), "连接成功",

                                                                        Toast.LENGTH_SHORT).show();

                                                }

                                        });

 

                                } catch (Exception e)

                                {

                                        e.printStackTrace();

                                }

                        };

                }.start();

        }

}

public class DeviceAdapter extends BaseAdapter

{

        private Context mContext;

        private ArrayList<BluetoothDevice> mDevices;

 

        public DeviceAdapter(Context context, ArrayList<BluetoothDevice> devices)

        {

                mDevices = devices;

                mContext = context;

        }

 

        @Override

        public int getCount()

        {

                return mDevices.size();

        }

 

        @Override

        public Object getItem(int position)

        {

                return mDevices.get(position);

        }

 

        @Override

        public long getItemId(int position)

        {

                return position;

        }

 

        @Override

        public View getView(int position, View convertView, ViewGroup parent)

        {

                ViewHolder holder;

                if (convertView == null)

                {

                        convertView = View.inflate(mContext, R.layout.item, null);

                        // item中就两个垂直排列的TextView

                        holder = new ViewHolder();

                        holder.mTvName = (TextView) convertView.findViewById(R.id.tv_name);

                        holder.mTvAddress = (TextView) convertView

                                        .findViewById(R.id.tv_address);

                        convertView.setTag(holder);

 

                } else

                {

                        holder = (ViewHolder) convertView.getTag();

                }

 

                BluetoothDevice device = mDevices.get(position);

                holder.mTvName.setText(device.getName());

                holder.mTvAddress.setText(device.getAddress());

 

                return convertView;

        }

 

        class ViewHolder

        {

                TextView mTvName;

                TextView mTvAddress;

        }

 

}
